17/// Internal node.
18///
19/// The layout of an internal node: `|ptr(children)/max(child keys)|...|ptr(children)|`.
20pub struct InternalNode<K, V> {
21    /// Children of the [`InternalNode`].
22    pub(super) children: Leaf<K, AtomicShared<Node<K, V>>>,
23
24    /// A child [`Node`] that has no upper key bound.
25    ///
26    /// It stores the maximum key in the node, and key-value pairs are firstly pushed to this
27    /// [`Node`] until split.
28    pub(super) unbounded_child: AtomicShared<Node<K, V>>,
29
30    /// On-going split operation.
31    split_op: StructuralChange<K, V>,
32
33    /// The latch protecting the [`InternalNode`].
34    latch: AtomicU8,
35
36    /// `wait_queue` for `latch`.
37    wait_queue: WaitQueue,
38}
39
40/// [`Locker`] holds exclusive ownership of a [`InternalNode`].
41pub(super) struct Locker<'n, K, V> {
42    internal_node: &'n InternalNode<K, V>,
43}

45/// [`StructuralChange`] stores intermediate results during a split operation.
46///
47/// `AtomicPtr` members may point to values under the protection of the [`Guard`] used for the
48/// split operation.
49struct StructuralChange<K, V> {
50    origin_node_key: AtomicPtr<K>,
51    origin_node: AtomicShared<Node<K, V>>,
52    low_key_node: AtomicShared<Node<K, V>>,
53    middle_key: AtomicPtr<K>,
54    high_key_node: AtomicShared<Node<K, V>>,
55}

57impl<K, V> InternalNode<K, V> {
58    /// Creates a new empty internal node.
59    #[inline]
60    pub(super) fn new() -> InternalNode<K, V> {
61        InternalNode {
62            children: Leaf::new(),
63            unbounded_child: AtomicShared::null(),
64            split_op: StructuralChange::default(),
65            latch: AtomicU8::new(Tag::None.into()),
66            wait_queue: WaitQueue::default(),
67        }
68    }
69
70    /// Clears the internal node.
71    #[inline]
72    pub(super) fn clear(&self, guard: &Guard) {
73        let scanner = Scanner::new(&self.children);
74        for (_, child) in scanner {
75            let child_ptr = child.load(Acquire, guard);
76            if let Some(child) = child_ptr.as_ref() {
77                child.clear(guard);
78            }
79        }
80        let unbounded_ptr = self.unbounded_child.load(Acquire, guard);
81        if let Some(unbounded) = unbounded_ptr.as_ref() {
82            unbounded.clear(guard);
83        }
84    }
85
86    /// Returns the depth of the node.
87    #[inline]
88    pub(super) fn depth(&self, depth: usize, guard: &Guard) -> usize {
89        let unbounded_ptr = self.unbounded_child.load(Relaxed, guard);
90        if let Some(unbounded_ref) = unbounded_ptr.as_ref() {
91            return unbounded_ref.depth(depth + 1, guard);
92        }
93        depth
94    }

96    /// Returns `true` if the [`InternalNode`] has retired.
97    #[inline]
98    pub(super) fn retired(&self) -> bool {
99        self.unbounded_child.tag(Acquire) == RETIRED
100    }
101
102    /// Waits for the lock on the [`InternalNode`] to be released.
103    #[inline]
104    pub(super) fn wait<D: DeriveAsyncWait>(&self, async_wait: &mut D) {
105        let waiter = || {
106            if self.latch.load(Acquire) == LOCKED.into() {
107                // The `InternalNode` is being split or locked.
108                return Err(());
109            }
110            Ok(())
111        };
112
113        if let Some(async_wait) = async_wait.derive() {
114            let _result = self.wait_queue.push_async_entry(async_wait, waiter);
115        } else {
116            let _result = self.wait_queue.wait_sync(waiter);
117        }
118    }
119
120    /// Tries to lock the [`InternalNode`].
121    fn try_lock(&self) -> bool {
122        self.latch
123            .compare_exchange(Tag::None.into(), LOCKED.into(), Acquire, Relaxed)
124            .is_ok()
125    }
126
127    /// Unlocks the [`InternalNode`].
128    fn unlock(&self) {
129        debug_assert_eq!(self.latch.load(Relaxed), LOCKED.into());
130        self.latch.store(Tag::None.into(), Release);
131        self.wait_queue.signal();
132    }
133
134    /// Retires itself.
135    fn retire(&self) {
136        debug_assert_eq!(self.latch.load(Relaxed), LOCKED.into());
137        self.latch.store(RETIRED.into(), Release);
138        self.wait_queue.signal();
139    }
140}

147    /// Searches for an entry containing the specified key.
148    #[inline]
149    pub(super) fn search_entry<'g, Q>(&self, key: &Q, guard: &'g Guard) -> Option<(&'g K, &'g V)>
150    where
151        K: 'g,
152        Q: Comparable<K> + ?Sized,
153    {
154        loop {
155            let (child, metadata) = self.children.min_greater_equal(key);
156            if let Some((_, child)) = child {
157                if let Some(child) = child.load(Acquire, guard).as_ref() {
158                    if self.children.validate(metadata) {
159                        // Data race resolution - see `LeafNode::search_entry`.
160                        return child.search_entry(key, guard);
161                    }
162                }
163            } else {
164                let unbounded_ptr = self.unbounded_child.load(Acquire, guard);
165                if let Some(unbounded) = unbounded_ptr.as_ref() {
166                    if self.children.validate(metadata) {
167                        return unbounded.search_entry(key, guard);
168                    }
169                } else {
170                    return None;
171                }
172            }
173        }
174    }

205    /// Returns the minimum key entry.
206    #[inline]
207    pub(super) fn min<'g>(&self, guard: &'g Guard) -> Option<Scanner<'g, K, V>> {
208        loop {
209            let mut retry = false;
210            let scanner = Scanner::new(&self.children);
211            let metadata = scanner.metadata();
212            for (_, child) in scanner {
213                let child_ptr = child.load(Acquire, guard);
214                if let Some(child) = child_ptr.as_ref() {
215                    if self.children.validate(metadata) {
216                        // Data race resolution - see `LeafNode::search_entry`.
217                        if let Some(scanner) = child.min(guard) {
218                            return Some(scanner);
219                        }
220                        continue;
221                    }
222                }
223                // It is not a hot loop - see `LeafNode::search_entry`.
224                retry = true;
225                break;
226            }
227            if retry {
228                continue;
229            }
230            let unbounded_ptr = self.unbounded_child.load(Acquire, guard);
231            if let Some(unbounded) = unbounded_ptr.as_ref() {
232                if self.children.validate(metadata) {
233                    return unbounded.min(guard);
234                }
235                continue;
236            }
237            return None;
238        }
239    }

241    /// Returns a [`Scanner`] pointing to an entry that is close enough to the entry with the
242    /// maximum key among those keys smaller than or equal to the given key.
243    ///
244    /// Returns `None` if all the keys in the [`InternalNode`] is equal to or greater than the
245    /// given key.
246    #[inline]
247    pub(super) fn max_le_appr<'g, Q>(&self, key: &Q, guard: &'g Guard) -> Option<Scanner<'g, K, V>>
248    where
249        K: 'g,
250        Q: Comparable<K> + ?Sized,
251    {
252        loop {
253            if let Some(scanner) = Scanner::max_less(&self.children, key) {
254                if let Some((_, child)) = scanner.get() {
255                    if let Some(child) = child.load(Acquire, guard).as_ref() {
256                        if self.children.validate(scanner.metadata()) {
257                            // Data race resolution - see `LeafNode::search_entry`.
258                            if let Some(scanner) = child.max_le_appr(key, guard) {
259                                return Some(scanner);
260                            }
261                            // Fallback.
262                            break;
263                        }
264                    }
265                    // It is not a hot loop - see `LeafNode::search_entry`.
266                    continue;
267                }
268            }
269            // Fallback.
270            break;
271        }
272
273        // Starts scanning from the minimum key.
274        let mut min_scanner = self.min(guard)?;
275        min_scanner.next();
276        loop {
277            if let Some((k, _)) = min_scanner.get() {
278                if key.compare(k).is_ge() {
279                    return Some(min_scanner);
280                }
281                break;
282            }
283            min_scanner = min_scanner.jump(None, guard)?;
284        }
285
286        None
287    }

289    /// Inserts a key-value pair.
290    #[inline]
291    pub(super) fn insert<D: DeriveAsyncWait>(
292        &self,
293        mut key: K,
294        mut val: V,
295        async_wait: &mut D,
296        guard: &Guard,
297    ) -> Result<InsertResult<K, V>, (K, V)> {
298        loop {
299            let (child, metadata) = self.children.min_greater_equal(&key);
300            if let Some((child_key, child)) = child {
301                let child_ptr = child.load(Acquire, guard);
302                if let Some(child_ref) = child_ptr.as_ref() {
303                    if self.children.validate(metadata) {
304                        // Data race resolution - see `LeafNode::search_entry`.
305                        let insert_result = child_ref.insert(key, val, async_wait, guard)?;
306                        match insert_result {
307                            InsertResult::Success
308                            | InsertResult::Duplicate(..)
309                            | InsertResult::Frozen(..) => return Ok(insert_result),
310                            InsertResult::Full(k, v) => {
311                                let split_result = self.split_node(
312                                    k,
313                                    v,
314                                    Some(child_key),
315                                    child_ptr,
316                                    child,
317                                    false,
318                                    async_wait,
319                                    guard,
320                                )?;
321                                if let InsertResult::Retry(k, v) = split_result {
322                                    key = k;
323                                    val = v;
324                                    continue;
325                                }
326                                return Ok(split_result);
327                            }
328                            InsertResult::Retired(k, v) => {
329                                debug_assert!(child_ref.retired());
330                                if self.coalesce(guard) == RemoveResult::Retired {
331                                    debug_assert!(self.retired());
332                                    return Ok(InsertResult::Retired(k, v));
333                                }
334                                return Err((k, v));
335                            }
336                            InsertResult::Retry(k, v) => {
337                                // `child` has been split, therefore it can be retried.
338                                if self.cleanup_link(&k, false, guard) {
339                                    key = k;
340                                    val = v;
341                                    continue;
342                                }
343                                return Ok(InsertResult::Retry(k, v));
344                            }
345                        };
346                    }
347                }
348                // It is not a hot loop - see `LeafNode::search_entry`.
349                continue;
350            }
351
352            let unbounded_ptr = self.unbounded_child.load(Acquire, guard);
353            if let Some(unbounded) = unbounded_ptr.as_ref() {
354                debug_assert!(unbounded_ptr.tag() == Tag::None);
355                if !self.children.validate(metadata) {
356                    continue;
357                }
358                let insert_result = unbounded.insert(key, val, async_wait, guard)?;
359                match insert_result {
360                    InsertResult::Success
361                    | InsertResult::Duplicate(..)
362                    | InsertResult::Frozen(..) => return Ok(insert_result),
363                    InsertResult::Full(k, v) => {
364                        let split_result = self.split_node(
365                            k,
366                            v,
367                            None,
368                            unbounded_ptr,
369                            &self.unbounded_child,
370                            false,
371                            async_wait,
372                            guard,
373                        )?;
374                        if let InsertResult::Retry(k, v) = split_result {
375                            key = k;
376                            val = v;
377                            continue;
378                        }
379                        return Ok(split_result);
380                    }
381                    InsertResult::Retired(k, v) => {
382                        debug_assert!(unbounded.retired());
383                        if self.coalesce(guard) == RemoveResult::Retired {
384                            debug_assert!(self.retired());
385                            return Ok(InsertResult::Retired(k, v));
386                        }
387                        return Err((k, v));
388                    }
389                    InsertResult::Retry(k, v) => {
390                        if self.cleanup_link(&k, false, guard) {
391                            key = k;
392                            val = v;
393                            continue;
394                        }
395                        return Ok(InsertResult::Retry(k, v));
396                    }
397                };
398            }
399            debug_assert!(unbounded_ptr.tag() == RETIRED);
400            return Ok(InsertResult::Retired(key, val));
401        }
402    }

404    /// Removes an entry associated with the given key.
405    ///
406    /// # Errors
407    ///
408    /// Returns an error if a retry is required with a Boolean flag indicating that an entry has been removed.
409    #[inline]
410    pub(super) fn remove_if<Q, F: FnMut(&V) -> bool, D>(
411        &self,
412        key: &Q,
413        condition: &mut F,
414        async_wait: &mut D,
415        guard: &Guard,
416    ) -> Result<RemoveResult, ()>
417    where
418        Q: Comparable<K> + ?Sized,
419        D: DeriveAsyncWait,
420    {
421        loop {
422            let (child, metadata) = self.children.min_greater_equal(key);
423            if let Some((_, child)) = child {
424                let child_ptr = child.load(Acquire, guard);
425                if let Some(child) = child_ptr.as_ref() {
426                    if self.children.validate(metadata) {
427                        // Data race resolution - see `LeafNode::search_entry`.
428                        let result =
429                            child.remove_if::<_, _, _>(key, condition, async_wait, guard)?;
430                        if result == RemoveResult::Cleanup {
431                            if self.cleanup_link(key, false, guard) {
432                                return Ok(RemoveResult::Success);
433                            }
434                            return Ok(RemoveResult::Cleanup);
435                        }
436                        if result == RemoveResult::Retired {
437                            return Ok(self.coalesce(guard));
438                        }
439                        return Ok(result);
440                    }
441                }
442                // It is not a hot loop - see `LeafNode::search_entry`.
443                continue;
444            }
445            let unbounded_ptr = self.unbounded_child.load(Acquire, guard);
446            if let Some(unbounded) = unbounded_ptr.as_ref() {
447                debug_assert!(unbounded_ptr.tag() == Tag::None);
448                if !self.children.validate(metadata) {
449                    // Data race resolution - see `LeafNode::search_entry`.
450                    continue;
451                }
452                let result = unbounded.remove_if::<_, _, _>(key, condition, async_wait, guard)?;
453                if result == RemoveResult::Cleanup {
454                    if self.cleanup_link(key, false, guard) {
455                        return Ok(RemoveResult::Success);
456                    }
457                    return Ok(RemoveResult::Cleanup);
458                }
459                if result == RemoveResult::Retired {
460                    return Ok(self.coalesce(guard));
461                }
462                return Ok(result);
463            }
464            return Ok(RemoveResult::Fail);
465        }
466    }

468    /// Removes a range of entries.
469    ///
470    /// Returns the number of remaining children.
471    #[allow(clippy::too_many_lines)]
472    #[inline]
473    pub(super) fn remove_range<'g, Q, R: RangeBounds<Q>, D: DeriveAsyncWait>(
474        &self,
475        range: &R,
476        start_unbounded: bool,
477        valid_lower_max_leaf: Option<&'g Leaf<K, V>>,
478        valid_upper_min_node: Option<&'g Node<K, V>>,
479        async_wait: &mut D,
480        guard: &'g Guard,
481    ) -> Result<usize, ()>
482    where
483        Q: Comparable<K> + ?Sized,
484    {
485        debug_assert!(valid_lower_max_leaf.is_none() || start_unbounded);
486        debug_assert!(valid_lower_max_leaf.is_none() || valid_upper_min_node.is_none());
487
488        let Some(_lock) = Locker::try_lock(self) else {
489            self.wait(async_wait);
490            return Err(());
491        };
492
493        let mut current_state = RemoveRangeState::Below;
494        let mut num_children = 1;
495        let mut lower_border = None;
496        let mut upper_border = None;
497
498        for (key, node) in Scanner::new(&self.children) {
499            current_state = current_state.next(key, range, start_unbounded);
500            match current_state {
501                RemoveRangeState::Below => {
502                    num_children += 1;
503                }
504                RemoveRangeState::MaybeBelow => {
505                    debug_assert!(!start_unbounded);
506                    num_children += 1;
507                    lower_border.replace((Some(key), node));
508                }
509                RemoveRangeState::FullyContained => {
510                    // There can be another thread inserting keys into the node, and this may
511                    // render those concurrent operations completely ineffective.
512                    self.children.remove_if(key, &mut |_| true);
513                    node.swap((None, Tag::None), AcqRel);
514                }
515                RemoveRangeState::MaybeAbove => {
516                    if valid_upper_min_node.is_some() {
517                        // `valid_upper_min_node` is not in this sub-tree.
518                        self.children.remove_if(key, &mut |_| true);
519                        node.swap((None, Tag::None), AcqRel);
520                    } else {
521                        num_children += 1;
522                        upper_border.replace(node);
523                    }
524                    break;
525                }
526            }
527        }
528
529        // Now, examine the unbounded child.
530        match current_state {
531            RemoveRangeState::Below => {
532                // The unbounded child is the only child, or all the children are below the range.
533                debug_assert!(lower_border.is_none() && upper_border.is_none());
534                if valid_upper_min_node.is_some() {
535                    lower_border.replace((None, &self.unbounded_child));
536                } else {
537                    upper_border.replace(&self.unbounded_child);
538                }
539            }
540            RemoveRangeState::MaybeBelow => {
541                debug_assert!(!start_unbounded);
542                debug_assert!(lower_border.is_some() && upper_border.is_none());
543                upper_border.replace(&self.unbounded_child);
544            }
545            RemoveRangeState::FullyContained => {
546                debug_assert!(upper_border.is_none());
547                upper_border.replace(&self.unbounded_child);
548            }
549            RemoveRangeState::MaybeAbove => {
550                debug_assert!(upper_border.is_some());
551            }
552        }
553
554        if let Some(lower_leaf) = valid_lower_max_leaf {
555            // It is currently in the middle of a recursive call: pass `lower_leaf` to connect leaves.
556            debug_assert!(start_unbounded && lower_border.is_none() && upper_border.is_some());
557            if let Some(upper_node) = upper_border.and_then(|n| n.load(Acquire, guard).as_ref()) {
558                upper_node.remove_range(range, true, Some(lower_leaf), None, async_wait, guard)?;
559            }
560        } else if let Some(upper_node) = valid_upper_min_node {
561            // Pass `upper_node` to the lower leaf to connect leaves, so that this method can be
562            // recursively invoked on `upper_node`.
563            debug_assert!(lower_border.is_some());
564            if let Some((Some(key), lower_node)) = lower_border {
565                self.children.remove_if(key, &mut |_| true);
566                self.unbounded_child
567                    .swap((lower_node.get_shared(Acquire, guard), Tag::None), AcqRel);
568                lower_node.swap((None, Tag::None), Release);
569            }
570            if let Some(lower_node) = self.unbounded_child.load(Acquire, guard).as_ref() {
571                lower_node.remove_range(
572                    range,
573                    start_unbounded,
574                    None,
575                    Some(upper_node),
576                    async_wait,
577                    guard,
578                )?;
579            }
580        } else {
581            let lower_node = lower_border.and_then(|n| n.1.load(Acquire, guard).as_ref());
582            let upper_node = upper_border.and_then(|n| n.load(Acquire, guard).as_ref());
583            match (lower_node, upper_node) {
584                (_, None) => (),
585                (None, Some(upper_node)) => {
586                    upper_node.remove_range(
587                        range,
588                        start_unbounded,
589                        None,
590                        None,
591                        async_wait,
592                        guard,
593                    )?;
594                }
595                (Some(lower_node), Some(upper_node)) => {
596                    debug_assert!(!ptr::eq(lower_node, upper_node));
597                    lower_node.remove_range(
598                        range,
599                        start_unbounded,
600                        None,
601                        Some(upper_node),
602                        async_wait,
603                        guard,
604                    )?;
605                }
606            }
607        }
608
609        Ok(num_children)
610    }

612    /// Splits a full node.
613    ///
614    /// # Errors
615    ///
616    /// Returns an error if a retry is required.
617    #[allow(clippy::too_many_arguments, clippy::too_many_lines)]
618    pub(super) fn split_node<D: DeriveAsyncWait>(
619        &self,
620        key: K,
621        val: V,
622        full_node_key: Option<&K>,
623        full_node_ptr: Ptr<Node<K, V>>,
624        full_node: &AtomicShared<Node<K, V>>,
625        root_split: bool,
626        async_wait: &mut D,
627        guard: &Guard,
628    ) -> Result<InsertResult<K, V>, (K, V)> {
629        let target = full_node_ptr.as_ref().unwrap();
630        if !self.try_lock() {
631            target.rollback(guard);
632            self.wait(async_wait);
633            return Err((key, val));
634        }
635        debug_assert!(!self.retired());
636
637        if full_node_ptr != full_node.load(Relaxed, guard) {
638            self.unlock();
639            target.rollback(guard);
640            return Err((key, val));
641        }
642
643        let prev = self
644            .split_op
645            .origin_node
646            .swap((full_node.get_shared(Relaxed, guard), Tag::None), Relaxed)
647            .0;
648        debug_assert!(prev.is_none());
649
650        if let Some(full_node_key) = full_node_key {
651            self.split_op
652                .origin_node_key
653                .store((full_node_key as *const K).cast_mut(), Relaxed);
654        }
655
656        let mut exit_guard = ExitGuard::new(true, |rollback| {
657            if rollback {
658                self.rollback(guard);
659            }
660        });
661        match target {
662            Node::Internal(full_internal_node) => {
663                // Copies nodes except for the known full node to the newly allocated internal node entries.
664                let internal_nodes = (
665                    Shared::new(Node::new_internal_node()),
666                    Shared::new(Node::new_internal_node()),
667                );
668                let Node::Internal(low_key_nodes) = internal_nodes.0.as_ref() else {
669                    unreachable!()
670                };
671                let Node::Internal(high_key_nodes) = internal_nodes.1.as_ref() else {
672                    unreachable!()
673                };
674
675                // Builds a list of valid nodes.
676                #[allow(clippy::type_complexity)]
677                let mut entry_array: [Option<(
678                    Option<&K>,
679                    AtomicShared<Node<K, V>>,
680                )>;
681                    DIMENSION.num_entries + 2] = Default::default();
682                let mut num_entries = 0;
683                let scanner = Scanner::new(&full_internal_node.children);
684                let recommended_boundary = Leaf::<K, V>::optimal_boundary(scanner.metadata());
685                for entry in scanner {
686                    if unsafe {
687                        full_internal_node
688                            .split_op
689                            .origin_node_key
690                            .load(Relaxed)
691                            .as_ref()
692                            .map_or_else(|| false, |key| entry.0 == key)
693                    } {
694                        let low_key_node_ptr = full_internal_node
695                            .split_op
696                            .low_key_node
697                            .load(Relaxed, guard);
698                        if !low_key_node_ptr.is_null() {
699                            entry_array[num_entries].replace((
700                                Some(unsafe {
701                                    full_internal_node
702                                        .split_op
703                                        .middle_key
704                                        .load(Relaxed)
705                                        .as_ref()
706                                        .unwrap()
707                                }),
708                                full_internal_node
709                                    .split_op
710                                    .low_key_node
711                                    .clone(Relaxed, guard),
712                            ));
713                            num_entries += 1;
714                        }
715                        let high_key_node_ptr = full_internal_node
716                            .split_op
717                            .high_key_node
718                            .load(Relaxed, guard);
719                        if !high_key_node_ptr.is_null() {
720                            entry_array[num_entries].replace((
721                                Some(entry.0),
722                                full_internal_node
723                                    .split_op
724                                    .high_key_node
725                                    .clone(Relaxed, guard),
726                            ));
727                            num_entries += 1;
728                        }
729                    } else {
730                        entry_array[num_entries]
731                            .replace((Some(entry.0), entry.1.clone(Acquire, guard)));
732                        num_entries += 1;
733                    }
734                }
735                if full_internal_node
736                    .split_op
737                    .origin_node_key
738                    .load(Relaxed)
739                    .is_null()
740                {
741                    // If the origin is an unbounded node, assign the high key node to the high key
742                    // node's unbounded.
743                    let low_key_node_ptr = full_internal_node
744                        .split_op
745                        .low_key_node
746                        .load(Relaxed, guard);
747                    if !low_key_node_ptr.is_null() {
748                        entry_array[num_entries].replace((
749                            Some(unsafe {
750                                full_internal_node
751                                    .split_op
752                                    .middle_key
753                                    .load(Relaxed)
754                                    .as_ref()
755                                    .unwrap()
756                            }),
757                            full_internal_node
758                                .split_op
759                                .low_key_node
760                                .clone(Relaxed, guard),
761                        ));
762                        num_entries += 1;
763                    }
764                    let high_key_node_ptr = full_internal_node
765                        .split_op
766                        .high_key_node
767                        .load(Relaxed, guard);
768                    if !high_key_node_ptr.is_null() {
769                        entry_array[num_entries].replace((
770                            None,
771                            full_internal_node
772                                .split_op
773                                .high_key_node
774                                .clone(Relaxed, guard),
775                        ));
776                        num_entries += 1;
777                    }
778                } else {
779                    // If the origin is a bounded node, assign the unbounded node to the high key
780                    // node's unbounded.
781                    entry_array[num_entries].replace((
782                        None,
783                        full_internal_node.unbounded_child.clone(Relaxed, guard),
784                    ));
785                    num_entries += 1;
786                }
787                debug_assert!(num_entries >= 2);
788
789                let low_key_node_array_size = recommended_boundary.min(num_entries - 1);
790                for (i, entry) in entry_array.iter().enumerate() {
791                    if let Some((k, v)) = entry {
792                        match (i + 1).cmp(&low_key_node_array_size) {
793                            Less => {
794                                low_key_nodes.children.insert_unchecked(
795                                    k.unwrap().clone(),
796                                    v.clone(Relaxed, guard),
797                                    i,
798                                );
799                            }
800                            Equal => {
801                                if let Some(&k) = k.as_ref() {
802                                    self.split_op
803                                        .middle_key
804                                        .store((k as *const K).cast_mut(), Relaxed);
805                                }
806                                low_key_nodes
807                                    .unbounded_child
808                                    .swap((v.get_shared(Relaxed, guard), Tag::None), Relaxed);
809                            }
810                            Greater => {
811                                if let Some(k) = k.cloned() {
812                                    high_key_nodes.children.insert_unchecked(
813                                        k,
814                                        v.clone(Relaxed, guard),
815                                        i - low_key_node_array_size,
816                                    );
817                                } else {
818                                    high_key_nodes
819                                        .unbounded_child
820                                        .swap((v.get_shared(Relaxed, guard), Tag::None), Relaxed);
821                                }
822                            }
823                        }
824                    } else {
825                        break;
826                    }
827                }
828
829                // Turns the new nodes into internal nodes.
830                self.split_op
831                    .low_key_node
832                    .swap((Some(internal_nodes.0), Tag::None), Relaxed);
833                self.split_op
834                    .high_key_node
835                    .swap((Some(internal_nodes.1), Tag::None), Relaxed);
836            }
837            Node::Leaf(full_leaf_node) => {
838                // Copies leaves except for the known full leaf to the newly allocated leaf node entries.
839                let leaf_nodes = (
840                    Shared::new(Node::new_leaf_node()),
841                    Shared::new(Node::new_leaf_node()),
842                );
843                let low_key_leaf_node = if let Node::Leaf(low_key_leaf_node) = leaf_nodes.0.as_ref()
844                {
845                    Some(low_key_leaf_node)
846                } else {
847                    None
848                };
849                let high_key_leaf_node =
850                    if let Node::Leaf(high_key_leaf_node) = &leaf_nodes.1.as_ref() {
851                        Some(high_key_leaf_node)
852                    } else {
853                        None
854                    };
855
856                self.split_op.middle_key.store(
857                    (full_leaf_node.split_leaf_node(
858                        low_key_leaf_node.unwrap(),
859                        high_key_leaf_node.unwrap(),
860                        guard,
861                    ) as *const K)
862                        .cast_mut(),
863                    Relaxed,
864                );
865
866                // Turns the new leaves into leaf nodes.
867                self.split_op
868                    .low_key_node
869                    .swap((Some(leaf_nodes.0), Tag::None), Relaxed);
870                self.split_op
871                    .high_key_node
872                    .swap((Some(leaf_nodes.1), Tag::None), Relaxed);
873            }
874        }
875
876        // Inserts the newly allocated internal nodes into the main array.
877        match self.children.insert(
878            unsafe {
879                self.split_op
880                    .middle_key
881                    .load(Relaxed)
882                    .as_ref()
883                    .unwrap()
884                    .clone()
885            },
886            self.split_op.low_key_node.clone(Relaxed, guard),
887        ) {
888            InsertResult::Success => (),
889            InsertResult::Duplicate(..) | InsertResult::Frozen(..) | InsertResult::Retry(..) => {
890                unreachable!()
891            }
892            InsertResult::Full(..) | InsertResult::Retired(..) => {
893                // Insertion failed: expects that the parent splits this node.
894                *exit_guard = false;
895                return Ok(InsertResult::Full(key, val));
896            }
897        }
898        *exit_guard = false;
899
900        // Replace the full node with the high-key node.
901        let unused_node = full_node
902            .swap(
903                (
904                    self.split_op.high_key_node.get_shared(Relaxed, guard),
905                    Tag::None,
906                ),
907                Release,
908            )
909            .0;
910
911        if root_split {
912            // Return without unlocking it.
913            return Ok(InsertResult::Retry(key, val));
914        }
915
916        // Unlock the node.
917        self.finish_split();
918
919        // Drop the deprecated nodes.
920        if let Some(unused_node) = unused_node {
921            // Clean up the split operation by committing it.
922            unused_node.commit(guard);
923            let _: bool = unused_node.release();
924        }
925
926        // Since a new node has been inserted, the caller can retry.
927        Ok(InsertResult::Retry(key, val))
928    }

930    /// Finishes splitting the [`InternalNode`].
931    #[inline]
932    pub(super) fn finish_split(&self) {
933        let origin = self.split_op.reset();
934        self.unlock();
935        self.wait_queue.signal();
936        origin.map(Shared::release);
937    }

939    /// Commits an on-going structural change recursively.
940    #[inline]
941    pub(super) fn commit(&self, guard: &Guard) {
942        let origin = self.split_op.reset();
943
944        // Mark the internal node retired to prevent further locking attempts.
945        self.retire();
946        if let Some(origin) = origin {
947            origin.commit(guard);
948            let _: bool = origin.release();
949        }
950    }

952    /// Rolls back the ongoing split operation recursively.
953    #[inline]
954    pub(super) fn rollback(&self, guard: &Guard) {
955        let origin = self.split_op.reset();
956        self.unlock();
957        if let Some(origin) = origin {
958            origin.rollback(guard);
959            let _: bool = origin.release();
960        }
961    }

963    /// Cleans up logically deleted leaves in the linked list.
964    ///
965    /// If the target leaf node does not exist in the sub-tree, returns `false`.
966    #[inline]
967    pub(super) fn cleanup_link<'g, Q>(&self, key: &Q, traverse_max: bool, guard: &'g Guard) -> bool
968    where
969        K: 'g,
970        Q: Comparable<K> + ?Sized,
971    {
972        if traverse_max {
973            // It just has to search for the maximum leaf node in the tree.
974            if let Some(unbounded) = self.unbounded_child.load(Acquire, guard).as_ref() {
975                return unbounded.cleanup_link(key, true, guard);
976            }
977        } else if let Some(child_scanner) = Scanner::max_less(&self.children, key) {
978            if let Some((_, child)) = child_scanner.get() {
979                if let Some(child) = child.load(Acquire, guard).as_ref() {
980                    return child.cleanup_link(key, true, guard);
981                }
982            }
983        }
984        false
985    }

987    /// Tries to coalesce nodes.
988    fn coalesce(&self, guard: &Guard) -> RemoveResult {
989        let mut node_deleted = false;
990        while let Some(lock) = Locker::try_lock(self) {
991            let mut max_key_entry = None;
992            for (key, node) in Scanner::new(&self.children) {
993                let node_ptr = node.load(Acquire, guard);
994                let node_ref = node_ptr.as_ref().unwrap();
995                if node_ref.retired() {
996                    let result = self.children.remove_if(key, &mut |_| true);
997                    debug_assert_ne!(result, RemoveResult::Fail);
998
999                    // Once the key is removed, it is safe to deallocate the node as the validation
1000                    // loop ensures the absence of readers.
1001                    if let Some(node) = node.swap((None, Tag::None), Release).0 {
1002                        let _: bool = node.release();
1003                        node_deleted = true;
1004                    }
1005                } else {
1006                    max_key_entry.replace((key, node));
1007                }
1008            }
1009
1010            // The unbounded node is replaced with the maximum key node if retired.
1011            let unbounded_ptr = self.unbounded_child.load(Acquire, guard);
1012            let fully_empty = if let Some(unbounded) = unbounded_ptr.as_ref() {
1013                if unbounded.retired() {
1014                    if let Some((key, max_key_child)) = max_key_entry {
1015                        if let Some(obsolete_node) = self
1016                            .unbounded_child
1017                            .swap(
1018                                (max_key_child.get_shared(Relaxed, guard), Tag::None),
1019                                Release,
1020                            )
1021                            .0
1022                        {
1023                            debug_assert!(obsolete_node.retired());
1024                            let _: bool = obsolete_node.release();
1025                            node_deleted = true;
1026                        }
1027                        let result = self.children.remove_if(key, &mut |_| true);
1028                        debug_assert_ne!(result, RemoveResult::Fail);
1029                        if let Some(node) = max_key_child.swap((None, Tag::None), Release).0 {
1030                            let _: bool = node.release();
1031                            node_deleted = true;
1032                        }
1033                        false
1034                    } else {
1035                        if let Some(obsolete_node) =
1036                            self.unbounded_child.swap((None, RETIRED), Release).0
1037                        {
1038                            debug_assert!(obsolete_node.retired());
1039                            let _: bool = obsolete_node.release();
1040                            node_deleted = true;
1041                        }
1042                        true
1043                    }
1044                } else {
1045                    false
1046                }
1047            } else {
1048                debug_assert!(unbounded_ptr.tag() == RETIRED);
1049                true
1050            };
1051
1052            if fully_empty {
1053                return RemoveResult::Retired;
1054            }
1055
1056            drop(lock);
1057            if !self.has_retired_node(guard) {
1058                break;
1059            }
1060        }
1061
1062        if node_deleted {
1063            RemoveResult::Cleanup
1064        } else {
1065            RemoveResult::Success
1066        }
1067    }

1069    /// Checks if the [`InternalNode`] has a retired [`Node`].
1070    fn has_retired_node(&self, guard: &Guard) -> bool {
1071        let mut has_valid_node = false;
1072        for entry in Scanner::new(&self.children) {
1073            let leaf_ptr = entry.1.load(Relaxed, guard);
1074            if let Some(leaf) = leaf_ptr.as_ref() {
1075                if leaf.retired() {
1076                    return true;
1077                }
1078                has_valid_node = true;
1079            }
1080        }
1081        if !has_valid_node {
1082            let unbounded_ptr = self.unbounded_child.load(Relaxed, guard);
1083            if let Some(unbounded) = unbounded_ptr.as_ref() {
1084                if unbounded.retired() {
1085                    return true;
1086                }
1087            }
1088        }
1089        false
1090    }
