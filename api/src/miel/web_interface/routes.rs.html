15async fn serve_static_file(path: warp::path::Tail) -> Result<impl Reply, Rejection> {
16    let path_str = path.as_str();
17
18    if let Some(asset) = WebUiAssets::get(path_str) {
19        let mime = mime_guess::from_path(path_str).first_or_octet_stream();
20
21        Ok(warp::reply::with_header(
22            asset.data.into_owned(), // Convert Cow to owned Vec<u8>
23            "content-type",
24            mime.as_ref(),
25        ))
26    } else {
27        Err(warp::reject::not_found())
28    }
29}

31async fn serve_index() -> Result<impl Reply, Rejection> {
32    if let Some(asset) = WebUiAssets::get("index.html") {
33        Ok(warp::reply::with_header(
34            asset.data.into_owned(), // Convert Cow to owned Vec<u8>
35            "content-type",
36            "text/html",
37        ))
38    } else {
39        Err(warp::reject::not_found())
40    }
41}

43/// GET /
44pub fn dashboard_route() -> impl Filter<Extract = impl Reply, Error = Rejection> + Clone {
45    let redirect_root = warp::path::end().and(warp::get()).and_then(serve_index);
46
47    //.map(|| warp::redirect::temporary(warp::http::Uri::from_static("/index.html")));
48
49    let static_files = warp::path::tail()
50        .and(warp::get())
51        .and_then(serve_static_file);
52
53    redirect_root.or(static_files)
54}

56/// GET /sessions
57pub fn list_sessions_route(
58    storage: Arc<dyn Storage + Send + Sync>,
59) -> impl Filter<Extract = impl Reply, Error = Rejection> + Clone {
60    warp::path!("api" / "sessions")
61        .and(warp::path::end())
62        .and(warp::get())
63        .and(warp::query::<SessionFilter>())
64        .and_then(move |filter: SessionFilter| {
65            let storage = storage.clone();
66            async move {
67                match storage.get_sessions(Some(filter)) {
68                    Ok(list) => {
69                        // Directly return storage sessions
70                        Ok::<_, Rejection>(warp::reply::with_status(
71                            warp::reply::json(&list),
72                            StatusCode::OK,
73                        ))
74                    }
75                    Err(_) => Ok::<_, Rejection>(warp::reply::with_status(
76                        warp::reply::json(&ApiError {
77                            message: "Failed to load sessions".to_string(),
78                        }),
79                        StatusCode::INTERNAL_SERVER_ERROR,
80                    )),
81                }
82            }
83        })
84}

86/// GET /sessions/:id/data
87pub fn get_session_data_route(
88    storage: Arc<dyn Storage + Send + Sync>,
89) -> impl Filter<Extract = impl Reply, Error = Rejection> + Clone {
90    warp::path!("api" / "sessions" / String / "data")
91        .and(warp::get())
92        .and_then(move |id_str: String| {
93            let storage = storage.clone();
94            async move {
95                let id = match Uuid::parse_str(&id_str) {
96                    Ok(u) => u,
97                    Err(_) => {
98                        let res = reply::with_status(
99                            reply::json(&ApiError {
100                                message: "Invalid session id".to_string(),
101                            }),
102                            StatusCode::BAD_REQUEST,
103                        )
104                        .into_response();
105                        return Ok::<_, Rejection>(res);
106                    }
107                };
108
109                match storage.get_session_data(id) {
110                    Ok(bytes) => {
111                        let res = reply::with_status(
112                            reply::with_header(bytes, "Content-Type", "application/octet-stream"),
113                            StatusCode::OK,
114                        )
115                        .into_response();
116                        Ok::<_, Rejection>(res)
117                    }
118                    Err(_) => {
119                        let res = reply::with_status(
120                            reply::json(&ApiError {
121                                message: "Session data not found".to_string(),
122                            }),
123                            StatusCode::NOT_FOUND,
124                        )
125                        .into_response();
126                        Ok::<_, Rejection>(res)
127                    }
128                }
129            }
130        })
131}
