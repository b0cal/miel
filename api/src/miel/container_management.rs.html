1//! Container management subsystem.
2//!
3//! This module exposes a minimal API to create and manage lightweight containers
4//! for honeypot services. The current implementation targets `systemd-nspawn`
5//! and focuses on simple lifecycle management and bookkeeping.
6//!
7//! Re-exports:
8//! - [`ContainerManager`]: main entry point to create/cleanup containers.
9//! - [`ContainerHandle`], [`ContainerStats`], [`Runtime`]: core types.
10//!
11//! Example (non-running):
12//! ```ignore
13//! use miel::container_management::{ContainerManager, ContainerStats};
14//!
15//! // Create a manager (will fail if `systemd-nspawn` is not available)
16//! let manager = ContainerManager::new()?;
17//! let stats: ContainerStats = manager.get_container_stats();
18//! println!("active: {}", stats.active_count);
19//! # Ok::<(), Box<dyn std::error::Error>>(())
20//! ```
21
22pub mod container_manager;
23pub mod obfuscation;
24pub mod types;
25
