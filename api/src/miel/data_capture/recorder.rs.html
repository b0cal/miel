1//! Stream recording orchestration for a single session.
2//!
3//! This module provides `StreamRecorder`, a small façade that ties together
4//! bidirectional TCP proxying (via `TcpCapture`) and best‑effort stdio/PTY
5//! snapshotting (via `StdioCapture`) for one honeypot session. It aggregates
6//! the resulting byte streams and metadata into `CaptureArtifacts` and persists
7//! them through the injected `Storage` implementation.
8//!
9//! Highlights
10//! - Full‑duplex TCP proxy with per‑direction buffering and timestamps
11//! - Graceful EOF propagation to avoid hangs (shutdown of the peer writer)
12//! - Optional PTY snapshot for stdout/stderr capture
13//! - Pluggable persistence through `Storage` (dependency injected)
14//! - Rich logging at TRACE/DEBUG/INFO
15//!
16//! Minimal usage
17//! ```ignore
18//! use std::sync::Arc;
19//! use tokio::net::TcpStream;
20//! use uuid::Uuid;
21//! use miel::data_capture::StreamRecorder;
22//! use miel::data_capture::CaptureArtifacts;
23//!
24//! // Your Storage implementation just needs to persist/retrieve artifacts.
25//! struct MyStorage;
26//! impl miel::data_capture::Storage for MyStorage {
27//!     fn save_capture_artifacts(&self, _a: &CaptureArtifacts) -> Result<(), miel::error_handling::types::StorageError> { Ok(()) }
28//!     fn get_capture_artifacts(&self, _id: Uuid) -> Result<CaptureArtifacts, miel::error_handling::types::StorageError> { todo!() }
29//! }
30//!
31//! # async fn run() -> Result<(), Box<dyn std::error::Error>> {
32//! let storage: Arc<dyn miel::data_capture::Storage> = Arc::new(MyStorage);
33//! let recorder = StreamRecorder::new(Uuid::new_v4(), storage);
34//!
35//! // Example sockets; in real code they come from your listener/container.
36//! let (client, server) = (TcpStream::connect("127.0.0.1:1").await?, TcpStream::connect("127.0.0.1:2").await?);
37//! let _ = recorder.start_tcp_proxy(client, server).await; // ignore errors in example
38//! let _ = recorder.finalize_capture(); // ignore errors in example
39//! # Ok(())
40//! # }
41//! ```

57/// Orchestrates network and stdio capture for a single session.
58///
59/// A `StreamRecorder` owns per‑session capture state and a reference to
60/// a pluggable [`Storage`] implementation. It exposes three operations:
61/// - [`StreamRecorder::start_tcp_proxy`]: full‑duplex forwarding between client and container
62///   sockets while recording both directions with timestamps.
63/// - [`StreamRecorder::start_stdio_capture`]: optional, best‑effort snapshot from a PTY handle
64///   (e.g. container stdout/stderr); safe to call zero or multiple times.
65/// - [`StreamRecorder::finalize_capture`]: aggregates all data into [`CaptureArtifacts`],
66///   persists them via [`Storage`], and returns the artifacts to the caller.
67///
68/// Logging
69/// - DEBUG/INFO for lifecycle milestones
70/// - TRACE for short (64‑byte) previews of captured chunks
71pub struct StreamRecorder {
72    /// Unique session identifier (used to correlate logs and persisted data).
73    session_id: Uuid,
74    /// TCP capture engine (both directions with timestamps).
75    tcp_capture: Arc<TcpCapture>,
76    /// Optional stdio/PTY snapshotter for the current session.
77    stdio_capture: Option<Arc<StdioCapture>>,
78    /// Pluggable persistence backend.
79    storage: Arc<dyn Storage + Send + Sync>,
80    /// Session start wall‑clock time (UTC), used to compute duration.
81    start_time: DateTime<Utc>,
82}
83
84impl StreamRecorder {
85    /// Creates a new `StreamRecorder` for the given `session_id` and `storage`.
86    ///
87    /// The recorder holds only lightweight buffers and references; it’s cheap to
88    /// construct and clone the underlying `Arc` values as needed by your orchestration.
89    pub fn new(session_id: Uuid, storage: Arc<dyn Storage + Send + Sync>) -> Self {
90        debug!("Stream recorder created for session {}", session_id);
91        Self {
92            session_id,
93            tcp_capture: Arc::new(TcpCapture::new(session_id)),
94            stdio_capture: None,
95            storage,
96            start_time: Utc::now(),
97        }
98    }
99
100    /// Starts a full‑duplex TCP proxy between the `client_stream` and the
101    /// `container_stream`, recording both directions.
102    ///
103    /// Behavior
104    /// - Forwards client→container and container→client using owned split halves.
105    /// - On EOF from one side, gracefully shuts down the opposite writer to wake
106    ///   the peer task and terminate without hangs.
107    /// - Records bytes and timestamps for both directions.
108    ///
109    /// Errors
110    /// - Returns [`CaptureError::TcpStreamError`] for read/write failures.
111    pub async fn start_tcp_proxy(
112        &self,
113        client_stream: TcpStream,
114        container_stream: TcpStream,
115    ) -> Result<(), CaptureError> {
116        debug!("Starting TCP proxy for session {}", self.session_id);
117        Arc::clone(&self.tcp_capture)
118            .proxy_and_record(client_stream, container_stream)
119            .await
120    }
121
122    /// Take a best‑effort PTY snapshot for stdio capture (non‑blocking where
123    /// possible) and appends results internally.
124    ///
125    /// Notes
126    /// - Safe to call multiple times; each call attempts a short read.
127    /// - It’s OK if the PTY isn’t readable yet (WouldBlock is ignored).
128    ///
129    /// Errors
130    /// - Returns [`CaptureError::StdioError`] for non‑recoverable IO failures.
131    pub fn start_stdio_capture(&mut self, pty_master: std::fs::File) -> Result<(), CaptureError> {
132        debug!("Starting stdio capture for session {}", self.session_id);
133        let cap = self
134            .stdio_capture
135            .get_or_insert_with(|| Arc::new(StdioCapture::new(self.session_id)))
136            .clone();
137        cap.capture_pty(pty_master)
138    }
139
140    /// Parse a unified activity log file and append its STDIN/STDOUT/STDERR
141    /// content to this recorder's stdio buffers.
142    pub fn parse_stdio_log_from_file<P: AsRef<Path>>(
143        &mut self,
144        path: P,
145    ) -> Result<(), CaptureError> {
146        debug!(
147            "Parsing stdio log from file for session {}",
148            self.session_id
149        );
150        let cap = self
151            .stdio_capture
152            .get_or_insert_with(|| Arc::new(StdioCapture::new(self.session_id)))
153            .clone();
154        cap.as_ref().capture_activity_log_from_path(path)
155    }
156
157    /// Aggregates TCP and stdio buffers into [`CaptureArtifacts`], computes
158    /// totals and duration, persists them via [`Storage`], and returns the
159    /// artifacts to the caller.
160    ///
161    /// Returns
162    /// - Persisted [`CaptureArtifacts`] for this session.
163    ///
164    /// Errors
165    /// - Returns [`CaptureError::StorageError`] if the storage backend fails to persist.
166    pub fn finalize_capture(&self) -> Result<CaptureArtifacts, CaptureError> {
167        let (c2s, s2c, tcp_ts) = self.tcp_capture.get_artifacts();
168
169        let (stdin, stdout, stderr, stdio_ts) = if let Some(ref stdio) = self.stdio_capture {
170            stdio.get_artifacts()
171        } else {
172            (Vec::new(), Vec::new(), Vec::new(), Vec::new())
173        };
174
175        let total_bytes: u64 =
176            (c2s.len() + s2c.len() + stdin.len() + stdout.len() + stderr.len()) as u64;
177        let duration = Utc::now() - self.start_time;
178
179        debug!(
180            "Capture finalized for session {}: {} bytes total (tcp_c2s={}, tcp_s2c={}, stdout={}, stderr={}), duration={:?}",
181            self.session_id,
182            total_bytes,
183            c2s.len(),
184            s2c.len(),
185            stdout.len(),
186            stderr.len(),
187            duration
188        );
189
190        let artifacts = CaptureArtifacts {
191            session_id: self.session_id,
192            tcp_client_to_container: c2s,
193            tcp_container_to_client: s2c,
194            stdio_stdin: String::from_utf8_lossy(&stdin).to_string(),
195            stdio_stdout: String::from_utf8_lossy(&stdout).to_string(),
196            stdio_stderr: String::from_utf8_lossy(&stderr).to_string(),
197            tcp_timestamps: tcp_ts,
198            stdio_timestamps: stdio_ts,
199            total_bytes,
200            duration,
201        };
202
203        self.storage
204            .save_capture_artifacts(&artifacts)
205            .map_err(|e| {
206                error!(
207                    "Failed to save capture artifacts for session {}: {}",
208                    self.session_id, e
209                );
210                CaptureError::StorageError(e)
211            })?;
212
213        debug!("Capture artifacts saved for session {}", self.session_id);
214        Ok(artifacts)
215    }
216}

289    async fn tcp_pair() -> std::io::Result<(TcpStream, TcpStream)> {
290        let listener = TcpListener::bind((std::net::Ipv4Addr::LOCALHOST, 0)).await?;
291        let addr = listener.local_addr()?;
292
293        let client = tokio::spawn(async move { TcpStream::connect(addr).await });
294        let (server_side, _) = listener.accept().await?;
295        let client = client.await.unwrap()?;
296        Ok((server_side, client))
297    }

299    #[tokio::test]
300    async fn tcp_proxy_captures_data() {
301        let _ = env_logger::builder().is_test(true).try_init();
302        let (client_server_side, mut client_outside) = tcp_pair().await.unwrap();
303        let (container_server_side, mut container_inside) = tcp_pair().await.unwrap();
304
305        let storage: Arc<dyn crate::storage::storage_trait::Storage + Send + Sync> =
306            Arc::new(MemStorage::new());
307        let recorder = Arc::new(StreamRecorder::new(Uuid::new_v4(), storage));
308
309        let rec2 = Arc::clone(&recorder);
310        let proxy = tokio::spawn(async move {
311            rec2.start_tcp_proxy(client_server_side, container_server_side)
312                .await
313        });
314
315        // Send payload client -> container
316        client_outside
317            .write_all(b"hello")
318            .await
319            .expect("write client->container");
320        client_outside.flush().await.ok();
321
322        let mut buf = [0u8; 16];
323        let n = container_inside
324            .read(&mut buf)
325            .await
326            .expect("read in container");
327        assert_eq!(&buf[..n], b"hello");
328
329        // Send response container -> client
330        container_inside
331            .write_all(b"pong")
332            .await
333            .expect("write container->client");
334        container_inside.flush().await.ok();
335
336        let mut rbuf = [0u8; 16];
337        let rn = client_outside
338            .read(&mut rbuf)
339            .await
340            .expect("read at client");
341        assert_eq!(&rbuf[..rn], b"pong");
342
343        // Gracefully shutdown both write halves to trigger EOF on proxy tasks
344        client_outside.shutdown().await.ok();
345        container_inside.shutdown().await.ok();
346
347        // Close streams to end proxy
348        drop(container_inside);
349        drop(client_outside);
350
351        // Wait for proxy completion with timeout to avoid hangs in CI
352        let res = tokio::time::timeout(std::time::Duration::from_secs(2), proxy).await;
353        match res {
354            Ok(join_res) => {
355                join_res.expect("proxy join").expect("proxy ok");
356            }
357            Err(_) => panic!("proxy task timed out"),
358        }
359
360        let artifacts = recorder.finalize_capture().expect("finalize ok");
361        assert!(artifacts.total_bytes >= 5);
362        assert!(artifacts
363            .tcp_timestamps
364            .iter()
365            .any(|(_, dir, n)| *dir == Direction::ClientToContainer && *n > 0));
366        assert!(artifacts
367            .tcp_timestamps
368            .iter()
369            .any(|(_, dir, n)| *dir == Direction::ContainerToClient && *n > 0));
370    }
