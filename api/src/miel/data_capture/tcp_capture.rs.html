1//! TCP capture: full‑duplex proxying while recording traffic and timestamps.
2//!
3//! This module exposes [`TcpCapture`], which forwards data between a client and
4//! a container/service while buffering both directions and recording per‑chunk
5//! timestamps. It is used by the higher‑level `StreamRecorder` façade.
6
7use std::io;
8use std::sync::{Arc, Mutex};
9
10use chrono::{DateTime, Utc};
11use log::trace;
12use tokio::io::{AsyncReadExt, AsyncWriteExt};
13use tokio::net::TcpStream;
14use tokio::task::JoinSet;
15use uuid::Uuid;
16
17use super::types::Direction;
18use crate::error_handling::types::CaptureError;
19
20type TcpTimestamps = Vec<(DateTime<Utc>, Direction, usize)>;
21type TcpArtifacts = (Vec<u8>, Vec<u8>, TcpTimestamps);
22
23#[derive(Debug)]
24/// Records TCP traffic for one session while acting as a transparent proxy.
25pub struct TcpCapture {
26    pub(crate) session_id: Uuid,
27    pub(crate) client_to_container: Mutex<Vec<u8>>,
28    pub(crate) container_to_client: Mutex<Vec<u8>>,
29    pub(crate) timestamps: Mutex<TcpTimestamps>,
30}
31
32impl TcpCapture {
33    /// Create a new `TcpCapture` instance for `session_id`.
34    pub fn new(session_id: Uuid) -> Self {
35        Self {
36            session_id,
37            client_to_container: Mutex::new(Vec::new()),
38            container_to_client: Mutex::new(Vec::new()),
39            timestamps: Mutex::new(Vec::new()),
40        }
41    }
42
43    /// Forward data in both directions while recording bytes and timestamps.
44    ///
45    /// Behavior
46    /// - Spawns two tasks (client→container and container→client).
47    /// - Gracefully propagates EOF by shutting down the opposite writer.
48    /// - Buffers payloads and pushes `(timestamp, direction, len)` entries.
49    ///
50    /// Errors
51    /// - Returns [`CaptureError::TcpStreamError`] for I/O or task failures.
52    pub async fn proxy_and_record(
53        self: Arc<Self>,
54        client_stream: TcpStream,
55        container_stream: TcpStream,
56    ) -> Result<(), CaptureError> {
57        let (cr, cw) = client_stream.into_split();
58        let (sr, sw) = container_stream.into_split();
59
60        trace!("[{:?}] starting tcp proxy", self.session_id);
61
62        let mut set = JoinSet::new();
63
64        // Client -> Container (read from client, write to container)
65        {
66            let this = Arc::clone(&self);
67            set.spawn(async move {
68                trace!("[{:?}] C->S task started", this.session_id);
69                let mut cr = cr;
70                let mut sw = sw; // forward to container writer
71                let mut buf = vec![0u8; 16 * 1024];
72                loop {
73                    let n = match cr.read(&mut buf).await {
74                        Ok(n) => n,
75                        Err(e) => break Err(CaptureError::TcpStreamError(e)),
76                    };
77                    if n == 0 {
78                        trace!(
79                            "[{:?}] C->S EOF; shutting down server writer",
80                            this.session_id
81                        );
82                        let _ = sw.shutdown().await; // signal EOF to container side
83                        break Ok(());
84                    }
85                    if let Err(e) = sw.write_all(&buf[..n]).await {
86                        break Err(CaptureError::TcpStreamError(e));
87                    }
88                    // record and trace
89                    {
90                        let mut data = this.client_to_container.lock().unwrap();
91                        data.extend_from_slice(&buf[..n]);
92                    }
93                    {
94                        let mut ts = this.timestamps.lock().unwrap();
95                        ts.push((Utc::now(), Direction::ClientToContainer, n));
96                    }
97                    let preview = &buf[..std::cmp::min(n, 64)];
98                    trace!(
99                        "[{:?}] captured C->S {} bytes: {}{}",
100                        this.session_id,
101                        n,
102                        String::from_utf8_lossy(preview),
103                        if n > 64 { " ..." } else { "" }
104                    );
105                }
106            });
107        }
108
109        // Container -> Client (read from container, write to client)
110        {
111            let this = Arc::clone(&self);
112            set.spawn(async move {
113                trace!("[{:?}] S->C task started", this.session_id);
114                let mut sr = sr;
115                let mut cw = cw; // forward to client writer
116                let mut buf = vec![0u8; 16 * 1024];
117                loop {
118                    let n = match sr.read(&mut buf).await {
119                        Ok(n) => n,
120                        Err(e) => break Err(CaptureError::TcpStreamError(e)),
121                    };
122                    if n == 0 {
123                        trace!(
124                            "[{:?}] S->C EOF; shutting down client writer",
125                            this.session_id
126                        );
127                        let _ = cw.shutdown().await; // signal EOF to client side
128                        break Ok(());
129                    }
130                    if let Err(e) = cw.write_all(&buf[..n]).await {
131                        break Err(CaptureError::TcpStreamError(e));
132                    }
133                    // record and trace
134                    {
135                        let mut data = this.container_to_client.lock().unwrap();
136                        data.extend_from_slice(&buf[..n]);
137                    }
138                    {
139                        let mut ts = this.timestamps.lock().unwrap();
140                        ts.push((Utc::now(), Direction::ContainerToClient, n));
141                    }
142                    let preview = &buf[..std::cmp::min(n, 64)];
143                    trace!(
144                        "[{:?}] captured S->C {} bytes: {}{}",
145                        this.session_id,
146                        n,
147                        String::from_utf8_lossy(preview),
148                        if n > 64 { " ..." } else { "" }
149                    );
150                }
151            });
152        }
153
154        while let Some(res) = set.join_next().await {
155            res.map_err(|e| CaptureError::TcpStreamError(io::Error::other(e)))??;
156        }
157
158        trace!("[{:?}] tcp proxy completed", self.session_id);
159        Ok(())
160    }
161
162    /// Return copies of client→container, container→client, and timestamp log.
163    pub fn get_artifacts(&self) -> TcpArtifacts {
164        let a = self.client_to_container.lock().unwrap().clone();
165        let b = self.container_to_client.lock().unwrap().clone();
166        let t = self.timestamps.lock().unwrap().clone();
167        (a, b, t)
168    }
169}
