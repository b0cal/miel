1//! Core types used by the container management subsystem.
2
3use chrono::{DateTime, Utc};
4use std::fs::File;
5use tokio::net::TcpStream;
6
7/// Aggregate counters describing the current and historical container state.
8#[derive(Debug, Clone)]
9pub struct ContainerStats {
10    /// Number of containers currently tracked as active.
11    pub active_count: usize,
12    /// Total number of containers successfully created since manager init.
13    pub total_created: u64,
14    /// Number of operations that failed (e.g., cleanup or start failures).
15    pub failed_count: u64,
16}
17
18/// Handle describing a specific container instance managed by the system.
19#[derive(Debug)]
20pub struct ContainerHandle {
21    /// Unique identifier for the container (e.g., `miel-<service>-<uuid>`).
22    pub id: String,
23    /// Logical service name that the container is running (e.g., "ssh").
24    pub service_name: String,
25    /// Fixed internal container port where the service listens.
26    pub port: u16,
27    /// Ephemeral host port mapped to the container's internal `port`.
28    pub host_port: u16, // ephemeral host port mapped to the container
29    /// Creation timestamp.
30    pub created_at: DateTime<Utc>,
31    /// Process handle for the runtime container process, if available.
32    pub process_handle: Option<tokio::process::Child>,
33    /// Optional PTY master used to capture stdio from the container.
34    pub pty_master: Option<File>,
35    /// Optional TCP socket associated to the service connection lifecycle.
36    pub tcp_socket: Option<TcpStream>,
37}
38
39// Implement Clone manually since tokio::process::Child and File don't implement Clone
40impl Clone for ContainerHandle {
41    fn clone(&self) -> Self {
42        ContainerHandle {
43            id: self.id.clone(),
44            service_name: self.service_name.clone(),
45            port: self.port,
46            host_port: self.host_port,
47            created_at: self.created_at,
48            process_handle: None, // Can't clone process handle
49            pty_master: None,     // Can't clone file handle
50            tcp_socket: None,     // Can't clone TCP stream
51        }
52    }
53}
