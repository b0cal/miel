17/// Orchestrates container lifecycle and bookkeeping for honeypot services.
18///
19/// The manager abstracts over a container runtime (currently [`Runtime::SystemdNspawn`])
20/// and maintains a registry of active containers along with simple counters.
21///
22/// Design notes:
23/// - Containers are created under `/tmp/miel-containers/<id>` and run with
24///   `systemd-nspawn --ephemeral` and `--private-network`.
25/// - A random ephemeral host port is allocated and mapped to the container's
26///   internal service port.
27/// - This is a minimal, best-effort implementation not meant for production isolation.
28#[derive(Clone)]
29pub struct ContainerManager {
30    runtime: Runtime,
31    active_containers: HashMap<String, ContainerHandle>,
32    stats: ContainerStats,
33}
34
35impl ContainerManager {
36    /// Creates a new `ContainerManager`.
37    ///
38    /// Returns an error if the configured runtime is not available on the host.
39    pub fn new() -> Result<Self, ContainerError> {
40        debug!("Initializing container manager");
41
42        // Check if systemd-nspawn is available, otherwise fail
43        if !Self::is_runtime_available() {
44            error!("Container runtime systemd-nspawn is not available");
45            return Err(ContainerError::RuntimeNotAvailable);
46        }
47
48        // Require root privileges: unprivileged nspawn with a plain directory tree is not supported
49        // on many systems and will implicitly enable private networking, breaking host-port binding.
50        if !Self::is_running_as_root() {
51            error!("Insufficient privileges: container manager requires root access");
52            return Err(ContainerError::StartFailed(
53                "systemd-nspawn requires root privileges for this setup. Please run the program with sudo.".to_string(),
54            ));
55        }
56
57        let manager = ContainerManager {
58            runtime: Runtime::SystemdNspawn,
59            active_containers: HashMap::new(),
60            stats: ContainerStats {
61                active_count: 0,
62                total_created: 0,
63                failed_count: 0,
64            },
65        };
66
67        info!(
68            "Container manager initialized with runtime: {:?}",
69            manager.runtime
70        );
71        Ok(manager)
72    }

92    fn is_running_as_root() -> bool {
93        let is_root = if let Ok(output) = std::process::Command::new("id").arg("-u").output() {
94            if output.status.success() {
95                if let Ok(s) = String::from_utf8(output.stdout) {
96                    s.trim() == "0"
97                } else {
98                    false
99                }
100            } else {
101                false
102            }
103        } else {
104            false
105        };
106
107        debug!("Root privilege check: {}", is_root);
108        is_root

111    /// Creates a new container for the given `service_config` and returns its handle.
112    ///
113    /// Side effects:
114    /// - Allocates an ephemeral host port (127.0.0.1) and maps it to the service port.
115    /// - Spawns a `systemd-nspawn` process with an ephemeral rootfs under `/tmp`.
116    /// - Updates internal stats and registry.
117    ///
118    /// Errors if the container cannot be prepared or started.
119    pub async fn create_container(
120        &mut self,
121        service_config: &ServiceConfig,
122    ) -> Result<ContainerHandle, ContainerError> {
123        let container_id = format!("miel-{}-{}", service_config.name, Uuid::new_v4());
124
125        debug!(
126            "Creating container {} for service {}",
127            container_id, service_config.name
128        );
129
130        // Use the runtime to create the container
131        let handle = match self.runtime {
132            Runtime::SystemdNspawn => {
133                debug!("Using systemd-nspawn runtime for container creation");
134                self.create_nspawn_container(service_config, &container_id)
135                    .await?
136            }
137        };
138
139        // Update stats
140        self.stats.total_created += 1;
141        self.stats.active_count += 1;
142
143        // Store the handle
144        self.active_containers
145            .insert(container_id.clone(), handle.clone());
146
147        info!(
148            "Container {} created for service {}",
149            container_id, service_config.name
150        );
151        Ok(handle)
152    }

154    /// Cleans up a specific container.
155    ///
156    /// Best-effort: attempts to kill the process, remove the registry entry,
157    /// decrement counters, and delete the temporary directory.
158    pub async fn cleanup_container(
159        &mut self,
160        mut handle: ContainerHandle,
161    ) -> Result<(), ContainerError> {
162        debug!("Cleaning up container: {}", handle.id);
163
164        // Kill the process if it's still running
165        if let Some(mut process) = handle.process_handle.take() {
166            debug!("Terminating process for container: {}", handle.id);
167            if let Err(e) = process.kill().await {
168                warn!("Failed to terminate container process {}: {}", handle.id, e);
169            } else {
170                debug!("Process terminated for container: {}", handle.id);
171            }
172        } else {
173            debug!("No active process found for container: {}", handle.id);
174        }
175
176        // Remove from active containers
177        self.active_containers.remove(&handle.id);
178        self.stats.active_count = self.stats.active_count.saturating_sub(1);
179
180        // Clean up container directory
181        let container_path = format!("/tmp/miel-containers/{}", handle.id);
182        debug!("Removing container directory: {}", container_path);
183        if let Err(e) = std::fs::remove_dir_all(&container_path) {
184            warn!(
185                "Failed to remove container directory {}: {}",
186                container_path, e
187            );
188        } else {
189            debug!("Container directory removed: {}", container_path);
190        }
191
192        debug!("Container cleanup completed: {}", handle.id);
193        Ok(())
194    }

196    /// Cleans up all tracked containers, continuing on errors and counting failures.
197    pub async fn cleanup_all_containers(&mut self) -> Result<(), ContainerError> {
198        let container_count = self.active_containers.len();
199        if container_count > 0 {
200            info!("Cleaning up {} active containers", container_count);
201        }
202
203        let container_handles: Vec<ContainerHandle> =
204            self.active_containers.values().cloned().collect();
205
206        for handle in container_handles {
207            if let Err(e) = self.cleanup_container(handle).await {
208                error!("Failed to cleanup container: {}", e);
209                self.stats.failed_count += 1;
210            }
211        }
212
213        self.active_containers.clear();
214        self.stats.active_count = 0;
215
216        if container_count > 0 {
217            info!(
218                "Container cleanup completed (failures: {})",
219                self.stats.failed_count
220            );
221        }
222        Ok(())
223    }

271    /// Creates a container using `systemd-nspawn` and returns its handle.
272    async fn create_nspawn_container(
273        &self,
274        service_config: &ServiceConfig,
275        container_id: &str,
276    ) -> Result<ContainerHandle, ContainerError> {
277        debug!("Creating systemd-nspawn container: {}", container_id);
278
279        // Create a basic container directory structure
280        let container_path = format!("/tmp/miel-containers/{}", container_id);
281        debug!("Preparing container directory: {}", container_path);
282        std::fs::create_dir_all(&container_path).map_err(|e| {
283            error!(
284                "Failed to create container directory {}: {}",
285                container_path, e
286            );
287            ContainerError::CreationFailed(format!("Failed to create container directory: {}", e))
288        })?;
289
290        // Create a basic rootfs structure
291        self.setup_container_rootfs(&container_path, service_config)
292            .await?;
293
294        // Apply obfuscation enhancements to the container
295        ObfuscationManager::setup_obfuscation(&container_path, &service_config.obfuscation)?;
296
297        // Prepare systemd-nspawn command
298        let mut cmd = Command::new("systemd-nspawn");
299        cmd.arg("--directory")
300            .arg(&container_path)
301            .arg("--ephemeral")
302            .arg("--bind-ro=/etc/resolv.conf");
303
304        // Create and bind the log directory so containers can write to it
305        let log_dir = "/tmp/miel-logs";
306        std::fs::create_dir_all(log_dir).map_err(|e| {
307            error!("Failed to create log directory {}: {}", log_dir, e);
308            ContainerError::CreationFailed(format!("Failed to create log directory: {}", e))
309        })?;
310
311        // Set permissions on the log directory to be writable by all users
312        #[cfg(unix)]
313        {
314            use std::os::unix::fs::PermissionsExt;
315            let mut perms = std::fs::metadata(log_dir)
316                .map_err(|e| {
317                    error!("Failed to get log directory metadata: {}", e);
318                    ContainerError::CreationFailed(format!(
319                        "Failed to get log directory metadata: {}",
320                        e
321                    ))
322                })?
323                .permissions();
324            perms.set_mode(0o777); // rwxrwxrwx - allow all users to write
325            std::fs::set_permissions(log_dir, perms).map_err(|e| {
326                error!("Failed to set log directory permissions: {}", e);
327                ContainerError::CreationFailed(format!(
328                    "Failed to set log directory permissions: {}",
329                    e
330                ))
331            })?;
332        }
333
334        // Bind the log directory so it's accessible from within the container
335        cmd.arg(format!("--bind={}", log_dir));
336        debug!("Log directory bound to container: {}", log_dir);
337
338        // Bind essential host dirs so common binaries and their libs are available
339        // inside the minimal rootfs. Only bind paths that exist on the host.
340        let mut bound_paths = 0;
341        for p in [
342            "/bin",
343            "/usr/bin",
344            "/sbin",
345            "/usr/sbin",
346            "/usr/libexec",
347            "/lib",
348            "/lib64",
349            "/usr/lib",
350            "/usr/lib64",
351        ]
352        .iter()
353        {
354            if Path::new(p).exists() {
355                cmd.arg(format!("--bind-ro={}", p));
356                bound_paths += 1;
357            }
358        }
359        debug!("Bound {} system paths to container", bound_paths);
360
361        cmd.arg(format!("--machine={}", container_id))
362            .stdin(Stdio::piped())
363            .stdout(Stdio::piped())
364            .stderr(Stdio::piped());
365
366        // Allocate an ephemeral host port for the service
367        let host_port = self.allocate_ephemeral_port(&service_config.protocol)?;
368        debug!("Allocated ephemeral port {} for container", host_port);
369
370        // Add the service command to run
371        cmd.arg("--").arg("/bin/sh").arg("-c");
372        let service_command = self.get_service_command(service_config, host_port, container_id);
373        cmd.arg(&service_command);
374
375        debug!(
376            "Starting systemd-nspawn process for container: {}",
377            container_id
378        );
379
380        // Start the process
381        let mut process = cmd.spawn().map_err(|e| {
382            error!("Failed to spawn container {}: {}", container_id, e);
383            ContainerError::StartFailed(format!("Failed to spawn container: {}", e))
384        })?;
385
386        // Capture stderr
387        if let Some(stderr) = process.stderr.take() {
388            let mut reader = BufReader::new(stderr).lines();
389            let cid = container_id.to_string();
390            tokio::spawn(async move {
391                while let Ok(Some(line)) = reader.next_line().await {
392                    debug!("[container:{}][stderr] {}", cid, line);
393                }
394                debug!("stderr monitoring ended for container: {}", cid);
395            });
396        }
397
398        // Create a PTY for stdio capture - now creates unified activity log
399        let pty_master = self.create_pty_master(container_id).ok();
400
401        // Capture stdout and redirect to unified log file
402        if let Some(stdout) = process.stdout.take() {
403            let log_path = format!("/tmp/miel-logs/container-{}-activity.log", container_id);
404            let mut reader = BufReader::new(stdout).lines();
405            let cid = container_id.to_string();
406            tokio::spawn(async move {
407                while let Ok(Some(line)) = reader.next_line().await {
408                    debug!("[container:{}][stdout] {}", cid, line);
409                    // Also write to the unified log file
410                    if let Ok(mut file) = std::fs::OpenOptions::new().append(true).open(&log_path) {
411                        use std::io::Write;
412                        let timestamp = Utc::now().format("%Y-%m-%d %H:%M:%S UTC");
413                        let _ = writeln!(file, "[{}] [CONTAINER] {}", timestamp, line);
414                        let _ = file.flush();
415                    }
416                }
417                debug!("stdout monitoring ended for container: {}", cid);
418            });
419        }
420
421        // Wait for the service to start up and establish a TCP connection
422        info!(
423            "Waiting for service to start and establishing TCP connection to container {}",
424            container_id
425        );
426        let tcp_socket = self
427            .establish_container_connection(host_port, container_id)
428            .await?;
429
430        let handle = ContainerHandle {
431            id: container_id.to_string(),
432            service_name: service_config.name.clone(),
433            port: service_config.port,
434            host_port,
435            created_at: Utc::now(),
436            process_handle: Some(process),
437            pty_master,
438            tcp_socket: Some(tcp_socket),
439        };
440
441        debug!(
442            "Successfully created nspawn container with TCP connection: {}",
443            container_id
444        );
445        Ok(handle)
446    }

448    /// Establishes a TCP connection to the container service with retry logic.
449    ///
450    /// This method waits for the service inside the container to start up and
451    /// then establishes a TCP connection that can be used for traffic forwarding.
452    async fn establish_container_connection(
453        &self,
454        host_port: u16,
455        container_id: &str,
456    ) -> Result<TcpStream, ContainerError> {
457        let max_retries = 30;
458        let mut retries = 0;
459        let target_addr = format!("127.0.0.1:{}", host_port);
460
461        debug!(
462            "Attempting to connect to container service at {}",
463            target_addr
464        );
465
466        while retries < max_retries {
467            match TcpStream::connect(&target_addr).await {
468                Ok(socket) => {
469                    info!(
470                        "Successfully established TCP connection to container {} on {}",
471                        container_id, target_addr
472                    );
473                    return Ok(socket);
474                }
475                Err(e) => {
476                    retries += 1;
477                    let wait_time = std::cmp::min(500 + (retries * 200), 3000); // Progressive backoff, cap at 3s
478                    debug!(
479                        "Connection attempt {}/{} failed for container {}: {} - retrying in {}ms",
480                        retries, max_retries, container_id, e, wait_time
481                    );
482                    tokio::time::sleep(tokio::time::Duration::from_millis(wait_time)).await;
483                }
484            }
485        }
486
487        Err(ContainerError::ConnectionFailed(format!(
488            "Failed to establish TCP connection to container {} at {} after {} attempts",
489            container_id, target_addr, max_retries
490        )))
491    }

496        container_path: &str,
497        service_config: &ServiceConfig,
498    ) -> Result<(), ContainerError> {

500
501        // Basic directory structure
502        let dirs = [
503            "bin",
504            "usr/bin",
505            "usr/local",
506            "usr/local/bin",
507            "usr/local/etc",
508            "sbin",
509            "usr/sbin",
510            "usr/libexec",
511            "usr/libexec/openssh",
512            "etc",
513            "etc/ssh",
514            "var",
515            "var/run",
516            "var/run/sshd",
517            "tmp",
518            "proc",
519            "sys",
520            "www",
521            "root",
522            "home",
523            "home/miel",
524            "usr/share",
525            "usr/share/empty.sshd",
526        ];
527
528        debug!("Creating {} directories in container", dirs.len());
529        for dir in &dirs {
530            let full_path = format!("{}/{}", container_path, dir);
531            std::fs::create_dir_all(&full_path).map_err(|e| {
532                error!("Failed to create directory {}: {}", full_path, e);
533                ContainerError::CreationFailed(format!("Failed to create dir {}: {}", dir, e))
534            })?;
535        }
536
537        // Copy essential binaries (simplified - in real implementation would use proper base image)
538        debug!("Copying essential binaries to container");
539        if let Err(e) = std::fs::copy("/bin/sh", format!("{}/bin/sh", container_path)) {
540            warn!("Failed to copy /bin/sh: {}", e);
541        } else {
542            debug!("Successfully copied /bin/sh");
543        }
544
545        // Provide minimal system files for sshd compatibility
546        // Credentials: miel:miel
547        let files = [
548            ("etc/passwd", "root:x:0:0:root:/root:/bin/sh\nsshd:x:74:74:sshd privilege separation user:/var/run/sshd:/bin/false\nmiel:x:1000:1000:miel User:/home/miel:/bin/sh\n"),
549            ("etc/group", "root:x:0:\nsshd:x:74:\nmiel:x:1000:\n"),
550            ("etc/shadow", "root:*:19000:0:99999:7:::\nsshd:*:19000:0:99999:7:::\nmiel:$6$JWGaRU6XKVQ3ONQJ$k/G2q0uMScsEKSjfMS6YteGEEGuOl2wdodXeU6QSQSsBXOC1wG0TPcmWvsa1elj7P4LCmKy9P1NcStmATA6h11:19000:0:99999:7:::\n"),
551            ("www/index.html", "<html><body><h1>Miel demo</h1><p>It works.</p></body></html>\n"),
552        ];
553
554        debug!("Creating {} system files", files.len());
555        for (path, content) in &files {
556            let full_path = format!("{}/{}", container_path, path);
557            if let Err(e) = std::fs::write(&full_path, content) {
558                warn!("Failed to write {}: {}", path, e);
559            } else {
560                debug!("Successfully created system file: {}", path);
561            }
562        }
563
564        // Create service script for the configuration
565        debug!("Creating service files for: {}", service_config.name);
566        let service_script = format!(
567            "#!/bin/sh\necho 'Starting {} service on port {}'\nwhile true; do\n    echo 'Service {} is running'\n    sleep 30\ndone\n",
568            service_config.name, service_config.port, service_config.name
569        );
570
571        std::fs::write(
572            format!("{}/usr/bin/service", container_path),
573            service_script,
574        )
575        .map_err(|e| {
576            error!("Failed to create service script: {}", e);
577            ContainerError::CreationFailed(format!("Failed to create service script: {}", e))
578        })?;
579
580        // Make service script executable
581        #[cfg(unix)]
582        {
583            use std::os::unix::fs::PermissionsExt;
584            debug!("Setting executable permissions for service script");
585            let mut perms = std::fs::metadata(format!("{}/usr/bin/service", container_path))
586                .map_err(|e| {
587                    error!("Failed to get script metadata: {}", e);
588                    ContainerError::CreationFailed(format!("Failed to get script metadata: {}", e))
589                })?
590                .permissions();
591            perms.set_mode(0o755);
592            std::fs::set_permissions(format!("{}/usr/bin/service", container_path), perms)
593                .map_err(|e| {
594                    error!("Failed to set script permissions: {}", e);
595                    ContainerError::CreationFailed(format!(
596                        "Failed to set script permissions: {}",
597                        e
598                    ))
599                })?;
600        }
601
602        debug!("Successfully set up container rootfs");
603        Ok(())
604    }

606    /// Returns the command line to run for a given `service_config`.
607    ///
608    /// For SSH services, this includes comprehensive logging configuration to capture
609    /// all session activity to the unified log file.
610    fn get_service_command(
611        &self,
612        service_config: &ServiceConfig,
613        host_port: u16,
614        container_id: &str,
615    ) -> String {
616        let log_path = format!("/tmp/miel-logs/container-{}-activity.log", container_id);
617        match service_config.name.as_str() {
618            "ssh" => {
619                let p = host_port;
620                format!(
621                    r#"
622                    /usr/bin/ssh-keygen -A >/dev/null 2>&1 || /bin/ssh-keygen -A >/dev/null 2>&1;
623                    cat > /etc/ssh/sshd_config << 'EOF'
624Port {p}
625ListenAddress 127.0.0.1
626Protocol 2
627HostKey /etc/ssh/ssh_host_rsa_key
628HostKey /etc/ssh/ssh_host_ecdsa_key
629HostKey /etc/ssh/ssh_host_ed25519_key
630UsePrivilegeSeparation no
631LogLevel VERBOSE
632PermitRootLogin no
633PasswordAuthentication yes
634UsePAM no
635Subsystem sftp /usr/libexec/openssh/sftp-server
636PidFile /var/run/sshd/sshd.pid
637EOF
638
639                    cat > /usr/local/bin/logged_shell << 'EOF'
640#!/bin/sh
641TS=$(date '+%Y-%m-%d %H:%M:%S UTC')
642echo "[$TS] [SSH-SESSION] Interactive shell session started" >> {log_path}
643export PATH="/tmp:$PATH"
644for cmd in ls cat pwd whoami id ps top netstat ss w who uname find grep awk sed tail head more less vi nano wget curl chmod chown mkdir rmdir rm cp mv ln tar gzip gunzip file which env printenv history mount df du free uptime lscpu lsblk ifconfig ip route iptables nmap nc telnet ping traceroute dig nslookup; do
645  if [ -x "/bin/$cmd" ] || [ -x "/usr/bin/$cmd" ] || [ -x "/sbin/$cmd" ] || [ -x "/usr/sbin/$cmd" ]; then
646    cat > "/tmp/$cmd" << CMD_EOF
647#!/bin/sh
648TS=\$(date '+%Y-%m-%d %H:%M:%S UTC')
649echo "[\$TS] [SSH] [STDIN] $cmd \$*" >> {log_path}
650OUT="/tmp/cmd_out_\$\$"; ERR="/tmp/cmd_err_\$\$"

783    /// Creates a unified log file to capture all shell activity from the container.
784    ///
785    /// This method creates a dedicated log file that will contain:
786    /// - Main container shell output
787    /// - SSH session activity (commands, output, interactions)
788    /// - Any other shell/PTY activity within the container
789    ///
790    /// The log file is created with appropriate permissions and can be read
791    /// to monitor all terminal activity happening inside the container.
792    fn create_pty_master(&self, container_id: &str) -> Result<File, ContainerError> {
793        // Create a dedicated log directory for container activity
794        let log_dir = "/tmp/miel-logs";
795        std::fs::create_dir_all(log_dir).map_err(|e| {
796            error!("Failed to create log directory {}: {}", log_dir, e);
797            ContainerError::CreationFailed(format!("Failed to create log directory: {}", e))
798        })?;
799
800        // Create a unified log file for all container shell activity
801        let log_path = format!("{}/container-{}-activity.log", log_dir, container_id);
802        debug!("Creating unified activity log at: {}", log_path);
803
804        let log_file = std::fs::OpenOptions::new()
805            .create(true)
806            .append(true)
807            .read(true)
808            .open(&log_path)
809            .map_err(|e| {
810                error!("Failed to create activity log {}: {}", log_path, e);
811                ContainerError::CreationFailed(format!("Failed to create activity log: {}", e))
812            })?;
813
814        // Set appropriate permissions to allow container to write
815        #[cfg(unix)]
816        {
817            use std::os::unix::fs::PermissionsExt;
818            let mut perms = log_file
819                .metadata()
820                .map_err(|e| {
821                    error!("Failed to get log file metadata: {}", e);
822                    ContainerError::CreationFailed(format!(
823                        "Failed to get log file metadata: {}",
824                        e
825                    ))
826                })?
827                .permissions();
828            perms.set_mode(0o666); // rw-rw-rw- - allow container to write
829            std::fs::set_permissions(&log_path, perms).map_err(|e| {
830                error!("Failed to set log file permissions: {}", e);
831                ContainerError::CreationFailed(format!("Failed to set log file permissions: {}", e))
832            })?;
833        }
834
835        // Write initial log header
836        use std::io::Write;
837        let mut file_writer = &log_file;
838        let timestamp = Utc::now().format("%Y-%m-%d %H:%M:%S UTC");
839        writeln!(
840            file_writer,
841            "=== Container {} Activity Log Started at {} ===",
842            container_id, timestamp
843        )
844        .map_err(|e| {
845            error!("Failed to write log header: {}", e);
846            ContainerError::CreationFailed(format!("Failed to write log header: {}", e))
847        })?;
848        file_writer.flush().map_err(|e| {
849            error!("Failed to flush log file: {}", e);
850            ContainerError::CreationFailed(format!("Failed to flush log file: {}", e))
851        })?;
852
853        info!(
854            "Successfully created unified activity log for container: {}",
855            container_id
856        );
857        Ok(log_file)
858    }

860    /// Allocates an ephemeral host port on 127.0.0.1 for the given protocol.
861    fn allocate_ephemeral_port(
862        &self,
863        protocol: &crate::configuration::types::Protocol,
864    ) -> Result<u16, ContainerError> {
865        let port = match protocol {
866            crate::configuration::types::Protocol::TCP => {
867                let listener = std::net::TcpListener::bind((std::net::Ipv4Addr::LOCALHOST, 0))
868                    .map_err(|e| {
869                        error!("Failed to allocate ephemeral TCP port: {}", e);
870                        ContainerError::CreationFailed(format!(
871                            "Failed to allocate ephemeral TCP port: {}",
872                            e
873                        ))
874                    })?;
875                let port = listener
876                    .local_addr()
877                    .map_err(|e| {
878                        error!("Failed to read local addr for ephemeral TCP port: {}", e);
879                        ContainerError::CreationFailed(format!(
880                            "Failed to read local addr for ephemeral TCP port: {}",
881                            e
882                        ))
883                    })?
884                    .port();
885                // Close the socket to free the port for nspawn to bind
886                drop(listener);
887                debug!("Allocated TCP port: {}", port);
888                port
889            }
890            crate::configuration::types::Protocol::UDP => {
891                let socket = std::net::UdpSocket::bind((std::net::Ipv4Addr::LOCALHOST, 0))
892                    .map_err(|e| {
893                        error!("Failed to allocate ephemeral UDP port: {}", e);
894                        ContainerError::CreationFailed(format!(
895                            "Failed to allocate ephemeral UDP port: {}",
896                            e
897                        ))
898                    })?;
899                let port = socket
900                    .local_addr()
901                    .map_err(|e| {
902                        error!("Failed to read local addr for ephemeral UDP port: {}", e);
903                        ContainerError::CreationFailed(format!(
904                            "Failed to read local addr for ephemeral UDP port: {}",
905                            e
906                        ))
907                    })?
908                    .port();
909                // Close the socket to free the port for nspawn to bind
910                drop(socket);
911                debug!("Allocated UDP port: {}", port);
912                port
913            }
914        };
915
916        debug!("Allocated ephemeral {:?} port: {}", protocol, port);
917        Ok(port)
918    }

921impl Drop for ContainerManager {
922    fn drop(&mut self) {
923        // Attempt to cleanup all containers when the manager is dropped
924        if !self.active_containers.is_empty() {
925            warn!(
926                "ContainerManager dropped with {} active containers - this may indicate a resource leak",
927                self.active_containers.len()
928            );
929
930            // Log the IDs of remaining containers for debugging
931            let remaining_ids: Vec<_> = self.active_containers.keys().collect();
932            warn!("Remaining container IDs: {:?}", remaining_ids);
933        } else {
934            debug!("ContainerManager dropped cleanly with no active containers");
935        }
936    }
