29/// Storage backend that uses SQLite via SeaORM.
30///
31/// Construct with [`DatabaseStorage::new`] to respect `MIEL_STORAGE_PATH` or
32/// [`DatabaseStorage::new_file`] for an explicit database path.
33pub struct DatabaseStorage {
34    conn: DatabaseConnection,
35}
36
37impl DatabaseStorage {
38    /// Default database filename used in the application's working directory
39    const DEFAULT_DB_FILE: &'static str = "miel.sqlite3";
40
41    /// Create or open the database using the configured storage path.
42    /// This method should be used when creating storage from application configuration.
43    pub async fn from_config_path<P: AsRef<Path>>(storage_path: P) -> Result<Self, StorageError> {
44        let db_path = storage_path.as_ref().join(Self::DEFAULT_DB_FILE);
45        debug!("Initializing database storage at: {}", db_path.display());
46        Self::new_file(db_path).await
47    }
48
49    /// Create or open the database using `MIEL_STORAGE_PATH` if set, otherwise in the current working directory
50    ///
51    /// This method respects the `MIEL_STORAGE_PATH` environment variable for both configuration
52    /// and environment-based deployments. The database file will be placed directly in the
53    /// specified storage path.
54    pub async fn new() -> Result<Self, StorageError> {
55        if let Ok(storage_dir) = env::var("MIEL_STORAGE_PATH") {
56            let db_path = Path::new(&storage_dir).join(Self::DEFAULT_DB_FILE);
57            if let Some(parent) = db_path.parent() {
58                std::fs::create_dir_all(parent).map_err(|_| StorageError::WriteFailed)?;
59            }
60            debug!(
61                "Using database storage from MIEL_STORAGE_PATH: {}",
62                db_path.display()
63            );
64            return Self::new_file(db_path).await;
65        }
66        let cwd = env::current_dir().map_err(|_| StorageError::ConnectionFailed)?;
67        let path = cwd.join(Self::DEFAULT_DB_FILE);
68        debug!(
69            "Using database storage in current directory: {}",
70            path.display()
71        );
72        Self::new_file(path).await
73    }
74
75    /// Create or open the database at the specified filesystem path.
76    ///
77    /// The file is created if missing; parent directories are ensured.
78    pub async fn new_file<P: AsRef<Path>>(path: P) -> Result<Self, StorageError> {
79        let path_ref = path.as_ref();
80        if let Some(parent) = path_ref.parent() {
81            std::fs::create_dir_all(parent).map_err(|_| StorageError::WriteFailed)?;
82        }
83        debug!("Connecting to SQLite database: {}", path_ref.display());
84        // DSN understood by sea-orm/sqlx driver; will create file if needed
85        let dsn = format!("sqlite://{}?mode=rwc", path_ref.to_string_lossy());
86
87        let conn = Database::connect(dsn).await.map_err(|e| {
88            error!("Failed to connect to database: {}", e);
89            StorageError::ConnectionFailed
90        })?;
91
92        // ensure foreign keys
93        conn.execute(Statement::from_string(
94            DbBackend::Sqlite,
95            "PRAGMA foreign_keys = ON".to_string(),
96        ))
97        .await
98        .map_err(|e| {
99            error!("Failed to enable foreign keys: {}", e);
100            StorageError::WriteFailed
101        })?;
102
103        // create schema
104        debug!("Creating database schema if not exists");
105        conn.execute(Statement::from_string(
106            DbBackend::Sqlite,
107            r#"
108            CREATE TABLE IF NOT EXISTS sessions (
109                id TEXT PRIMARY KEY,
110                service_name TEXT NOT NULL,
111                client_addr TEXT NOT NULL,
112                start_time TEXT NOT NULL,
113                end_time TEXT,
114                container_id TEXT,
115                bytes_transferred INTEGER NOT NULL,
116                status TEXT NOT NULL
117            );
118        "#
119            .to_string(),
120        ))
121        .await
122        .map_err(|e| {
123            error!("Failed to create sessions table: {}", e);
124            StorageError::WriteFailed
125        })?;
126
127        conn.execute(Statement::from_string(
128            DbBackend::Sqlite,
129            r#"
130            CREATE TABLE IF NOT EXISTS interactions (
131                id INTEGER PRIMARY KEY AUTOINCREMENT,
132                session_id TEXT NOT NULL,
133                data BLOB NOT NULL,
134                FOREIGN KEY(session_id) REFERENCES sessions(id) ON DELETE CASCADE
135            );
136        "#
137            .to_string(),
138        ))
139        .await
140        .map_err(|e| {
141            error!("Failed to create interactions table: {}", e);
142            StorageError::WriteFailed
143        })?;
144
145        conn.execute(Statement::from_string(
146            DbBackend::Sqlite,
147            r#"
148            CREATE TABLE IF NOT EXISTS artifacts (
149                session_id TEXT PRIMARY KEY,
150                json TEXT NOT NULL,
151                FOREIGN KEY(session_id) REFERENCES sessions(id) ON DELETE CASCADE
152            );
153        "#
154            .to_string(),
155        ))
156        .await
157        .map_err(|e| {
158            error!("Failed to create artifacts table: {}", e);
159            StorageError::WriteFailed
160        })?;
161
162        debug!("Database storage initialized successfully");
163        Ok(Self { conn })
164    }

166    fn session_to_model(s: &Session) -> session::ActiveModel {
167        session::ActiveModel {
168            id: Set(s.id.to_string()),
169            service_name: Set(s.service_name.clone()),
170            client_addr: Set(s.client_addr.to_string()),
171            start_time: Set(s.start_time.to_rfc3339()),
172            end_time: Set(s.end_time.map(|t| t.to_rfc3339())),
173            container_id: Set(s.container_id.clone()),
174            bytes_transferred: Set(s.bytes_transferred as i64),
175            status: Set(match s.status {
176                crate::session_management::SessionStatus::Pending => "Pending",
177                crate::session_management::SessionStatus::Active => "Active",
178                crate::session_management::SessionStatus::Completed => "Completed",
179                crate::session_management::SessionStatus::Error => "Error",
180            }
181            .to_string()),
182        }
183    }
184
185    fn from_session_model(m: session::Model) -> Result<Session, StorageError> {
186        let status = match m.status.as_str() {
187            "Pending" => crate::session_management::SessionStatus::Pending,
188            "Active" => crate::session_management::SessionStatus::Active,
189            "Completed" => crate::session_management::SessionStatus::Completed,
190            _ => crate::session_management::SessionStatus::Error,
191        };
192        Ok(Session {
193            id: Uuid::parse_str(&m.id).map_err(|_| StorageError::ReadFailed)?,
194            service_name: m.service_name,
195            client_addr: m
196                .client_addr
197                .parse()
198                .map_err(|_| StorageError::ReadFailed)?,
199            start_time: DateTime::parse_from_rfc3339(&m.start_time)
200                .map_err(|_| StorageError::ReadFailed)?
201                .with_timezone(&Utc),
202            end_time: match m.end_time {
203                Some(s) => Some(
204                    DateTime::parse_from_rfc3339(&s)
205                        .map_err(|_| StorageError::ReadFailed)?
206                        .with_timezone(&Utc),
207                ),
208                None => None,
209            },
210            container_id: m.container_id,
211            bytes_transferred: m.bytes_transferred as u64,
212            status,
213        })
214    }

217impl Storage for DatabaseStorage {
218    fn save_session(&self, session_obj: &Session) -> Result<(), StorageError> {
219        let mut am = Self::session_to_model(session_obj);
220        let conn = self.conn.clone();
221        let session_id = session_obj.id.to_string();
222
223        tokio::task::block_in_place(|| {
224            tokio::runtime::Handle::current().block_on(async {
225                match session::Entity::find_by_id(session_id)
226                    .one(&conn)
227                    .await
228                    .map_err(|e| {
229                        error!("DB read error in save_session find_by_id: {}", e);
230                        StorageError::ReadFailed
231                    })? {
232                    Some(existing) => {
233                        am.id = Set(existing.id);
234                        am.update(&conn).await.map_err(|e| {
235                            error!("DB write error in save_session update: {}", e);
236                            StorageError::WriteFailed
237                        })?;
238                        info!("Updated a session");
239                    }
240                    None => {
241                        session::Entity::insert(am).exec(&conn).await.map_err(|e| {
242                            error!("DB write error in save_session insert exec: {}", e);
243                            StorageError::WriteFailed
244                        })?;
245                        info!("Inserted a session");
246                    }
247                }
248                Ok(())
249            })
250        })
251    }
252
253    fn get_sessions(&self, filter: Option<SessionFilter>) -> Result<Vec<Session>, StorageError> {
254        let conn = self.conn.clone();
255
256        tokio::task::block_in_place(|| {
257            tokio::runtime::Handle::current().block_on(async {
258                let mut query = session::Entity::find();
259                if filter.is_some() {
260                    debug!("Applying session filter");
261                }
262                if let Some(f) = filter {
263                    let mut cond = Condition::all();
264                    if let Some(name) = f.service_name {
265                        cond = cond.add(session::Column::ServiceName.eq(name));
266                    }
267                    if let Some(start) = f.start_date {
268                        cond = cond.add(session::Column::StartTime.gte(start.to_rfc3339()));
269                    }
270                    if let Some(end) = f.end_date {
271                        let coalesce = Func::coalesce([
272                            Expr::col(session::Column::EndTime).into(),
273                            Expr::col(session::Column::StartTime).into(),
274                        ]);
275                        cond = cond.add(Expr::expr(coalesce).lte(end.to_rfc3339()));
276                    }
277                    if let Some(ip) = f.client_addr {
278                        cond = cond.add(session::Column::ClientAddr.like(format!("{}:%", ip)));
279                    }
280                    if let Some(st) = f.status {
281                        let s = match st {
282                            crate::session_management::SessionStatus::Pending => "Pending",
283                            crate::session_management::SessionStatus::Active => "Active",
284                            crate::session_management::SessionStatus::Completed => "Completed",
285                            crate::session_management::SessionStatus::Error => "Error",
286                        };
287                        cond = cond.add(session::Column::Status.eq(s));
288                    }
289                    query = query.filter(cond);
290                }
291                let rows = query.all(&conn).await.map_err(|e| {
292                    error!("DB read error in get_sessions: {}", e);
293                    StorageError::ReadFailed
294                })?;
295                debug!("Fetched {} session rows", rows.len());
296                rows.into_iter().map(Self::from_session_model).collect()
297            })
298        })
299    }

301    fn save_interaction(&self, session_id: Uuid, data: &[u8]) -> Result<(), StorageError> {
302        let conn = self.conn.clone();
303        let data = data.to_vec();
304
305        tokio::task::block_in_place(|| {
306            tokio::runtime::Handle::current().block_on(async {
307                let am = inter::ActiveModel {
308                    session_id: Set(session_id.to_string()),
309                    data: Set(data.clone()),
310                    ..Default::default()
311                };
312                am.insert(&conn).await.map_err(|e| {
313                    error!("DB write error in save_interaction insert: {}", e);
314                    StorageError::WriteFailed
315                })?;
316                debug!("Inserted an interaction ({} bytes)", data.len());
317                Ok(())
318            })
319        })
320    }
321
322    fn get_session_data(&self, session_id: Uuid) -> Result<Vec<u8>, StorageError> {
323        let conn = self.conn.clone();
324
325        tokio::task::block_in_place(|| {
326            tokio::runtime::Handle::current().block_on(async {
327                let mut out = Vec::new();
328                let rows = inter::Entity::find()
329                    .filter(inter::Column::SessionId.eq(session_id.to_string()))
330                    .order_by_asc(inter::Column::Id)
331                    .all(&conn)
332                    .await
333                    .map_err(|e| {
334                        error!("DB read error in get_session_data: {}", e);
335                        StorageError::ReadFailed
336                    })?;
337                let mut chunks = 0usize;
338                for r in rows {
339                    out.extend_from_slice(&r.data);
340                    chunks += 1;
341                }
342                debug!(
343                    "Concatenated {} interaction chunk(s), total {} bytes",
344                    chunks,
345                    out.len()
346                );
347                Ok(out)
348            })
349        })
350    }
351
352    fn cleanup_old_sessions(&self, older_than: DateTime<Utc>) -> Result<usize, StorageError> {
353        let conn = self.conn.clone();
354
355        tokio::task::block_in_place(|| {
356            tokio::runtime::Handle::current().block_on(async {
357                let cutoff = older_than.to_rfc3339();
358                let coalesce = Func::coalesce([
359                    Expr::col(session::Column::EndTime).into(),
360                    Expr::col(session::Column::StartTime).into(),
361                ]);
362                let cond = Expr::expr(coalesce).lt(cutoff.clone());
363                let res = session::Entity::delete_many()
364                    .filter(cond)
365                    .exec(&conn)
366                    .await
367                    .map_err(|e| {
368                        error!("DB write error in cleanup_old_sessions delete_many: {}", e);
369                        StorageError::WriteFailed
370                    })?;
371                info!(
372                    "Deleted {} session(s) older than {}",
373                    res.rows_affected, cutoff
374                );
375                Ok(res.rows_affected as usize)
376            })
377        })
378    }

380    fn save_capture_artifacts(&self, artifacts: &CaptureArtifacts) -> Result<(), StorageError> {
381        let conn = self.conn.clone();
382        let id = artifacts.session_id.to_string();
383        let json = serde_json::to_string(artifacts).map_err(|_| StorageError::WriteFailed)?;
384
385        tokio::task::block_in_place(|| {
386            tokio::runtime::Handle::current().block_on(async {
387                match art::Entity::find_by_id(id.clone())
388                    .one(&conn)
389                    .await
390                    .map_err(|e| {
391                        error!("DB read error in save_capture_artifacts find_by_id: {}", e);
392                        StorageError::ReadFailed
393                    })? {
394                    Some(_) => {
395                        let am = art::ActiveModel {
396                            session_id: Set(id.clone()),
397                            json: Set(json),
398                        };
399                        am.update(&conn).await.map_err(|e| {
400                            error!("DB write error in save_capture_artifacts update: {}", e);
401                            StorageError::WriteFailed
402                        })?;
403                        info!("Updated artifacts for a session");
404                    }
405                    None => {
406                        let am = art::ActiveModel {
407                            session_id: Set(id.clone()),
408                            json: Set(json),
409                        };
410                        art::Entity::insert(am).exec(&conn).await.map_err(|e| {
411                            error!(
412                                "DB write error in save_capture_artifacts insert exec: {}",
413                                e
414                            );
415                            StorageError::WriteFailed
416                        })?;
417                        info!("Inserted artifacts for a session");
418                    }
419                }
420                Ok(())
421            })
422        })
423    }
424
425    fn get_capture_artifacts(&self, session_id: Uuid) -> Result<CaptureArtifacts, StorageError> {
426        let conn = self.conn.clone();
427
428        tokio::task::block_in_place(|| {
429            tokio::runtime::Handle::current().block_on(async {
430                let id = session_id.to_string();
431                let m = art::Entity::find_by_id(id.clone())
432                    .one(&conn)
433                    .await
434                    .map_err(|e| {
435                        error!("DB read error in get_capture_artifacts find_by_id: {}", e);
436                        StorageError::ReadFailed
437                    })?
438                    .ok_or(StorageError::ReadFailed)?;
439                let artifacts: CaptureArtifacts =
440                    serde_json::from_str(&m.json).map_err(|_| StorageError::ReadFailed)?;
441                debug!("Loaded artifacts from database");
442                Ok(artifacts)
443            })
444        })
445    }

454    async fn temp_db() -> DatabaseStorage {
455        let dir = TempDir::new().unwrap();
456        let path = dir.path().join("test.sqlite3");

458        Box::leak(Box::new(dir));

462    #[tokio::test(flavor = "multi_thread")]
463    async fn test_db_session_and_filter() {
464        let storage = temp_db().await;
465        let now = Utc::now();
466        let s1 = Session {
467            id: Uuid::new_v4(),
468            service_name: "ssh".into(),
469            client_addr: "127.0.0.1:2222".parse().unwrap(),
470            start_time: now,
471            end_time: Some(now),
472            container_id: None,
473            bytes_transferred: 100,
474            status: SessionStatus::Completed,
475        };
476        storage.save_session(&s1).unwrap();
477        let all = storage.get_sessions(None).unwrap();
478        assert_eq!(all.len(), 1);
479        let filtered = storage
480            .get_sessions(Some(SessionFilter {
481                service_name: Some("ssh".into()),
482                ..Default::default()
483            }))
484            .unwrap();
485        assert_eq!(filtered.len(), 1);
486        let none = storage
487            .get_sessions(Some(SessionFilter {
488                service_name: Some("http".into()),
489                ..Default::default()
490            }))
491            .unwrap();
492        assert_eq!(none.len(), 0);
493    }
