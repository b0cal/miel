1//! Filesystem-backed storage implementation.
2//!
3//! This backend persists sessions as human-readable text files, interactions as
4//! binary blobs, and artifacts in a per-session directory tree. It's intended
5//! for easy inspection and simple deployments. The root directory can be
6//! provided via `MIEL_STORAGE_PATH` or specified explicitly.

23/// Replace potentially sensitive path components (such as UUID-like names) with a redacted token.
24fn sanitize_path(p: &Path) -> String {
25    let mut parts: Vec<String> = Vec::new();
26    for comp in p.components() {
27        let s = comp.as_os_str().to_string_lossy().to_string();
28        // Basic heuristic: redact components that look like UUIDs or files that embed them
29        let redact = s.len() >= 8 && s.contains('-');
30        if redact {
31            parts.push("[redacted]".to_string());
32        } else {
33            parts.push(s);
34        }
35    }
36    parts.join("/")
37}

39/// Storage backend that writes data to the local filesystem.
40///
41/// Layout under the root directory:
42/// - `sessions/` — one `<uuid>.session` file per session (KV text)
43/// - `interactions/` — one `<uuid>.bin` concatenating interaction bytes
44/// - `artifacts/<uuid>/` — per-session directory with `*.bin`, `*.csv`, and `meta.txt`
45pub struct FileStorage {
46    base_path: PathBuf,
47    session_index: Mutex<HashMap<Uuid, PathBuf>>, // maps id to session file path
48    artifacts_path: PathBuf,
49}

51impl FileStorage {
52    /// Create a `FileStorage` rooted at `base_path`.
53    ///
54    /// The necessary subdirectories are created if missing.
55    pub fn new<P: AsRef<Path>>(base_path: P) -> Result<Self, StorageError> {
56        let base_path = base_path.as_ref().to_path_buf();
57        let sessions_dir = base_path.join("sessions");
58        let interactions_dir = base_path.join("interactions");
59        let artifacts_path = base_path.join("artifacts");
60
61        fs::create_dir_all(&sessions_dir).map_err(|e| {
62            error!(
63                "Failed to create sessions directory {}: {}",
64                sessions_dir.display(),
65                e
66            );
67            StorageError::WriteFailed
68        })?;
69        fs::create_dir_all(&interactions_dir).map_err(|e| {
70            error!(
71                "Failed to create interactions directory {}: {}",
72                interactions_dir.display(),
73                e
74            );
75            StorageError::WriteFailed
76        })?;
77        fs::create_dir_all(&artifacts_path).map_err(|e| {
78            error!(
79                "Failed to create artifacts directory {}: {}",
80                artifacts_path.display(),
81                e
82            );
83            StorageError::WriteFailed
84        })?;
85
86        debug!("File storage initialized at: {}", base_path.display());
87
88        Ok(Self {
89            base_path,
90            session_index: Mutex::new(HashMap::new()),
91            artifacts_path,
92        })
93    }

104    ///

106    /// and environment-based deployments. File storage will be created in a 'file_storage'

111            debug!(

118            error!("Failed to determine current directory: {}", e);
119            StorageError::ReadFailed
120        })?;
121        let base_path = cwd.join("file_storage");
122        debug!(
123            "Using file storage in current directory: {}",
124            base_path.display()
125        );
126        Self::new(base_path)

143    fn write_session_file(&self, session: &Session) -> Result<(), StorageError> {
144        let path = self.session_file_path(session.id);
145        let mut f = File::create(&path).map_err(|e| {
146            error!(
147                "Failed to create session file {}: {}",
148                sanitize_path(&path),
149                e
150            );
151            StorageError::WriteFailed
152        })?;
153        // Simple KV text format
154        writeln!(f, "id: {}", session.id).map_err(|e| {
155            error!("Failed to write session file {}: {}", path.display(), e);
156            StorageError::WriteFailed
157        })?;
158        writeln!(f, "service_name: {}", session.service_name).map_err(|e| {
159            error!("Failed to write session file {}: {}", path.display(), e);
160            StorageError::WriteFailed
161        })?;
162        writeln!(f, "client_addr: {}", session.client_addr).map_err(|e| {
163            error!("Failed to write session file {}: {}", path.display(), e);
164            StorageError::WriteFailed
165        })?;
166        writeln!(f, "start_time: {}", session.start_time.to_rfc3339()).map_err(|e| {
167            error!("Failed to write session file {}: {}", path.display(), e);
168            StorageError::WriteFailed
169        })?;
170        if let Some(end) = session.end_time {
171            writeln!(f, "end_time: {}", end.to_rfc3339()).map_err(|e| {
172                error!("Failed to write session file {}: {}", path.display(), e);
173                StorageError::WriteFailed
174            })?;
175        } else {
176            writeln!(f, "end_time: none").map_err(|e| {
177                error!("Failed to write session file {}: {}", path.display(), e);
178                StorageError::WriteFailed
179            })?;
180        }
181        if let Some(ref cid) = session.container_id {
182            writeln!(f, "container_id: {}", cid).map_err(|e| {
183                error!("Failed to write session file {}: {}", path.display(), e);
184                StorageError::WriteFailed
185            })?;
186        } else {
187            writeln!(f, "container_id: none").map_err(|e| {
188                error!("Failed to write session file {}: {}", path.display(), e);
189                StorageError::WriteFailed
190            })?;
191        }
192        writeln!(f, "bytes_transferred: {}", session.bytes_transferred).map_err(|e| {
193            error!("Failed to write session file {}: {}", path.display(), e);
194            StorageError::WriteFailed
195        })?;
196        let status_str = match session.status {
197            crate::session_management::SessionStatus::Pending => "Pending",
198            crate::session_management::SessionStatus::Active => "Active",
199            crate::session_management::SessionStatus::Completed => "Completed",
200            crate::session_management::SessionStatus::Error => "Error",
201        };
202        writeln!(f, "status: {}", status_str).map_err(|e| {
203            error!("Failed to write session file {}: {}", path.display(), e);
204            StorageError::WriteFailed
205        })?;
206
207        // update index
208        if let Ok(mut idx) = self.session_index.lock() {
209            idx.insert(session.id, path.clone());
210        }
211        debug!("Session data saved successfully");
212        Ok(())
213    }

215    fn parse_session_file(&self, path: &Path) -> Result<Session, StorageError> {
216        let mut content = String::new();
217        File::open(path)
218            .and_then(|mut f| f.read_to_string(&mut content))
219            .map_err(|e| {
220                error!("Failed to read session file {}: {}", sanitize_path(path), e);
221                StorageError::ReadFailed
222            })?;
223        let mut map: HashMap<String, String> = HashMap::new();
224        for line in content.lines() {
225            if let Some((k, v)) = line.split_once(": ") {
226                map.insert(k.trim().to_string(), v.trim().to_string());
227            }
228        }
229        let id = map
230            .get("id")
231            .ok_or_else(|| {
232                error!("Missing id in session file {}", path.display());
233                StorageError::ReadFailed
234            })?
235            .parse::<Uuid>()
236            .map_err(|e| {
237                error!("Invalid UUID in {}: {}", path.display(), e);
238                StorageError::ReadFailed
239            })?;
240        let service_name = map.remove("service_name").ok_or_else(|| {
241            error!("Missing service_name in {}", path.display());
242            StorageError::ReadFailed
243        })?;
244        let client_addr_str = map.remove("client_addr").ok_or_else(|| {
245            error!("Missing client_addr in {}", path.display());
246            StorageError::ReadFailed
247        })?;
248        let client_addr = client_addr_str.parse().map_err(|e| {
249            error!("Invalid client_addr in {}: {}", path.display(), e);
250            StorageError::ReadFailed
251        })?;
252        let start_time = map
253            .remove("start_time")
254            .ok_or_else(|| {
255                error!("Missing start_time in {}", path.display());
256                StorageError::ReadFailed
257            })
258            .and_then(|s| {
259                DateTime::parse_from_rfc3339(&s)
260                    .map(|dt| dt.with_timezone(&Utc))
261                    .map_err(|e| {
262                        error!("Invalid start_time in {}: {}", path.display(), e);
263                        StorageError::ReadFailed
264                    })
265            })?;
266        let end_time = map
267            .remove("end_time")
268            .and_then(|s| if s == "none" { None } else { Some(s) })
269            .map(|s| {
270                DateTime::parse_from_rfc3339(&s)
271                    .map(|dt| dt.with_timezone(&Utc))
272                    .map_err(|e| {
273                        error!("Invalid end_time in {}: {}", path.display(), e);
274                        StorageError::ReadFailed
275                    })
276            })
277            .transpose()?;
278        let container_id =
279            map.remove("container_id")
280                .and_then(|s| if s == "none" { None } else { Some(s) });
281        let bytes_transferred = map
282            .remove("bytes_transferred")
283            .ok_or_else(|| {
284                error!("Missing bytes_transferred in {}", path.display());
285                StorageError::ReadFailed
286            })?
287            .parse::<u64>()
288            .map_err(|e| {
289                error!("Invalid bytes_transferred in {}: {}", path.display(), e);
290                StorageError::ReadFailed
291            })?;
292        let status = match map
293            .remove("status")
294            .ok_or_else(|| {
295                error!("Missing status in {}", path.display());
296                StorageError::ReadFailed
297            })?
298            .as_str()
299        {
300            "Pending" => crate::session_management::SessionStatus::Pending,
301            "Active" => crate::session_management::SessionStatus::Active,
302            "Completed" => crate::session_management::SessionStatus::Completed,
303            _ => crate::session_management::SessionStatus::Error,
304        };
305        debug!("Session data parsed successfully");
306        Ok(Session {
307            id,
308            service_name,
309            client_addr,
310            start_time,
311            end_time,
312            container_id,
313            bytes_transferred,
314            status,
315        })
316    }

324    fn get_sessions(&self, filter: Option<SessionFilter>) -> Result<Vec<Session>, StorageError> {
325        let mut sessions = Vec::new();
326        for entry in fs::read_dir(self.sessions_dir()).map_err(|e| {
327            error!("Failed to read sessions dir: {}", e);
328            StorageError::ReadFailed
329        })? {
330            let entry = entry.map_err(|e| {
331                error!("Dir entry error: {}", e);
332                StorageError::ReadFailed
333            })?;
334            let path = entry.path();
335            if path.extension().and_then(|s| s.to_str()) == Some("session") {
336                if let Ok(sess) = self.parse_session_file(&path) {
337                    sessions.push(sess);
338                }
339            }
340        }
341        let original_len = sessions.len();
342        if let Some(f) = filter {
343            sessions.retain(|s| {
344                // service_name
345                if let Some(ref name) = f.service_name {
346                    if &s.service_name != name {
347                        return false;
348                    }
349                }
350                // time range
351                if let Some(start) = f.start_date {
352                    if s.start_time < start {
353                        return false;
354                    }
355                }
356                if let Some(end) = f.end_date {
357                    if s.end_time.unwrap_or(s.start_time) > end {
358                        return false;
359                    }
360                }
361                // client ip
362                if let Some(ip) = f.client_addr {
363                    if s.client_addr.ip() != ip {
364                        return false;
365                    }
366                }
367                // status
368                if let Some(ref st) = f.status {
369                    if &s.status != st {
370                        return false;
371                    }
372                }
373                true
374            });
375        }
376        debug!(
377            "Retrieved {} sessions ({} after filtering)",
378            original_len,
379            sessions.len()
380        );
381        Ok(sessions)
382    }

384    fn save_interaction(&self, session_id: Uuid, data: &[u8]) -> Result<(), StorageError> {
385        let path = self.interactions_dir().join(format!("{}.bin", session_id));
386        let mut f = OpenOptions::new()
387            .create(true)
388            .append(true)
389            .open(&path)
390            .map_err(|e| {
391                error!("Open append failed {}: {}", sanitize_path(&path), e);
392                StorageError::WriteFailed
393            })?;
394        f.write_all(data).map_err(|e| {
395            error!("Write failed {}: {}", sanitize_path(&path), e);
396            StorageError::WriteFailed
397        })?;
398        debug!("Interaction data appended ({} bytes)", data.len());
399        Ok(())
400    }

415    fn cleanup_old_sessions(&self, older_than: DateTime<Utc>) -> Result<usize, StorageError> {
416        let mut removed = 0usize;
417        for entry in fs::read_dir(self.sessions_dir()).map_err(|e| {
418            error!("Failed to read sessions dir: {}", e);
419            StorageError::ReadFailed
420        })? {
421            let entry = entry.map_err(|e| {
422                error!("Dir entry error: {}", e);
423                StorageError::ReadFailed
424            })?;
425            let path = entry.path();
426            if path.extension().and_then(|s| s.to_str()) != Some("session") {
427                continue;
428            }
429            if let Ok(sess) = self.parse_session_file(&path) {
430                let ts = sess.end_time.unwrap_or(sess.start_time);
431                if ts < older_than {
432                    let _ = fs::remove_file(&path);
433                    let _ =
434                        fs::remove_file(self.interactions_dir().join(format!("{}.bin", sess.id)));
435                    let _ = fs::remove_dir_all(self.artifacts_dir_for(sess.id));
436                    removed += 1;
437                }
438            }
439        }
440        info!("Cleaned up {} expired sessions", removed);
441        Ok(removed)
442    }

624    fn get_capture_artifacts(&self, session_id: Uuid) -> Result<CaptureArtifacts, StorageError> {
625        let dir = self.artifacts_dir_for(session_id);
626        let read_bin = |name: &str| -> Result<Vec<u8>, StorageError> {
627            let p = dir.join(name);
628            let mut buf = Vec::new();
629            File::open(&p)
630                .and_then(|mut f| f.read_to_end(&mut buf))
631                .map_err(|e| {
632                    error!("Read failed {}: {}", sanitize_path(&p), e);
633                    StorageError::ReadFailed
634                })?;
635            Ok(buf)
636        };
637        let tcp_client_to_container = read_bin("tcp_client_to_container.bin")?;
638        let tcp_container_to_client = read_bin("tcp_container_to_client.bin")?;
639        let stdio_stdin = read_bin("stdio_stdin.bin")?;
640        let stdio_stdout = read_bin("stdio_stdout.bin")?;
641        let stdio_stderr = read_bin("stdio_stderr.bin")?;
642
643        // parse timestamps
644        let mut tcp_timestamps: Vec<(DateTime<Utc>, Direction, usize)> = Vec::new();
645        let mut s = String::new();
646        File::open(dir.join("tcp_timestamps.csv"))
647            .and_then(|mut f| f.read_to_string(&mut s))
648            .map_err(|e| {
649                error!(
650                    "Read failed {}: {}",
651                    dir.join("tcp_timestamps.csv").display(),
652                    e
653                );
654                StorageError::ReadFailed
655            })?;
656        for line in s.lines() {
657            let parts: Vec<_> = line.split(',').map(|x| x.trim()).collect();
658            if parts.len() != 3 {
659                continue;
660            }
661            let ts = DateTime::parse_from_rfc3339(parts[0])
662                .map_err(|e| {
663                    error!("Invalid timestamp in tcp_timestamps.csv: {}", e);
664                    StorageError::ReadFailed
665                })?
666                .with_timezone(&Utc);
667            let dirn = match parts[1] {
668                "C2S" => Direction::ClientToContainer,
669                "S2C" => Direction::ContainerToClient,
670                _ => Direction::ClientToContainer,
671            };
672            let sz = parts[2].parse::<usize>().map_err(|e| {
673                error!("Invalid size in tcp_timestamps.csv: {}", e);
674                StorageError::ReadFailed
675            })?;
676            tcp_timestamps.push((ts, dirn, sz));
677        }
678
679        let mut stdio_timestamps: Vec<(DateTime<Utc>, StdioStream, usize)> = Vec::new();
680        s.clear();
681        File::open(dir.join("stdio_timestamps.csv"))
682            .and_then(|mut f| f.read_to_string(&mut s))
683            .map_err(|e| {
684                error!(
685                    "Read failed {}: {}",
686                    dir.join("stdio_timestamps.csv").display(),
687                    e
688                );
689                StorageError::ReadFailed
690            })?;
691        for line in s.lines() {
692            let parts: Vec<_> = line.split(',').map(|x| x.trim()).collect();
693            if parts.len() != 3 {
694                continue;
695            }
696            let ts = DateTime::parse_from_rfc3339(parts[0])
697                .map_err(|e| {
698                    error!("Invalid timestamp in stdio_timestamps.csv: {}", e);
699                    StorageError::ReadFailed
700                })?
701                .with_timezone(&Utc);
702            let stream = match parts[1] {
703                "Stdin" => StdioStream::Stdin,
704                "Stdout" => StdioStream::Stdout,
705                _ => StdioStream::Stderr,
706            };
707            let sz = parts[2].parse::<usize>().map_err(|e| {
708                error!("Invalid size in stdio_timestamps.csv: {}", e);
709                StorageError::ReadFailed
710            })?;
711            stdio_timestamps.push((ts, stream, sz));
712        }
713
714        // meta
715        s.clear();
716        File::open(dir.join("meta.txt"))
717            .and_then(|mut f| f.read_to_string(&mut s))
718            .map_err(|e| {
719                error!("Read failed {}: {}", dir.join("meta.txt").display(), e);
720                StorageError::ReadFailed
721            })?;
722        let mut total_bytes = 0u64;
723        let mut duration_secs = 0i64;
724        for line in s.lines() {
725            if let Some((k, v)) = line.split_once(':') {
726                let k = k.trim();
727                let v = v.trim();
728                match k {
729                    "total_bytes" => total_bytes = v.parse().unwrap_or(0),
730                    "duration_secs" => duration_secs = v.parse().unwrap_or(0),
731                    _ => {}
732                }
733            }
734        }
735        let duration = chrono::Duration::seconds(duration_secs);
736        debug!("Session artifacts parsed successfully");
737
738        Ok(CaptureArtifacts {
739            session_id,
740            tcp_client_to_container,
741            tcp_container_to_client,
742            stdio_stdin: String::from_utf8_lossy(&stdio_stdin).to_string(),
743            stdio_stdout: String::from_utf8_lossy(&stdio_stdout).to_string(),
744            stdio_stderr: String::from_utf8_lossy(&stdio_stderr).to_string(),
745            tcp_timestamps,
746            stdio_timestamps,
747            total_bytes,
748            duration,
749        })
750    }
