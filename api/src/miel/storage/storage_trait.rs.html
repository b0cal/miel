1//! Storage Trait
2//!
3//! This module defines the `Storage` trait, which provides an interface for session and artifact storage backends.
4//!
5//! Implementors of this trait are responsible for:
6//! - Persisting and retrieving session data
7//! - Managing interaction data
8//! - Handling capture artifacts
9//! - Cleaning up old sessions
10//!
11//! All methods return a `Result` to handle potential storage errors.
12
13use crate::data_capture::CaptureArtifacts;
14use crate::error_handling::types::StorageError;
15use crate::session::Session;
16use crate::storage::types::SessionFilter;
17use chrono::{DateTime, Utc};
18use uuid::Uuid;
19
20/// The `Storage` trait defines the interface for session and artifact storage backends.
21///
22/// Implementors of this trait are responsible for persisting and retrieving session data,
23/// interaction data, and capture artifacts, as well as cleaning up old sessions.
24///
25/// All methods return a `Result` to handle potential storage errors.
26pub trait Storage: Send + Sync {
27    /// Saves a session to the storage backend.
28    ///
29    /// - `session` - The `Session` to be saved.
30    fn save_session(&self, session: &Session) -> Result<(), StorageError>;
31
32    /// Retrieves sessions, optionally filtered.
33    fn get_sessions(&self, filter: Option<SessionFilter>) -> Result<Vec<Session>, StorageError>;
34
35    /// Saves interaction data for a given session.
36    fn save_interaction(&self, session_id: Uuid, data: &[u8]) -> Result<(), StorageError>;
37
38    /// Retrieves all interaction data for a given session.
39    fn get_session_data(&self, session_id: Uuid) -> Result<Vec<u8>, StorageError>;
40
41    /// Cleans up sessions older than the specified date and time.
42    fn cleanup_old_sessions(&self, older_than: DateTime<Utc>) -> Result<usize, StorageError>;
43
44    /// Saves capture artifacts to the storage backend.
45    fn save_capture_artifacts(&self, artifacts: &CaptureArtifacts) -> Result<(), StorageError>;
46
47    /// Retrieves capture artifacts for a given session.
48    fn get_capture_artifacts(&self, session_id: Uuid) -> Result<CaptureArtifacts, StorageError>;
49}
